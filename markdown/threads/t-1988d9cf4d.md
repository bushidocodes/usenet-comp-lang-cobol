	"someone says the leap year code is too tricky"
	/* leap years are divisible by 4, so a remainder is not a leap year */
	if (y % 4)
		return 0;
	return 1;

REVISION 3:
	"one of our customers said 2000 isn't a leap year, because it's
	 a century"
	/* leap years are divisible by 4, so a remainder is not a leap year */
	if (y % 4)
		return 0;

		return 1;

REVISION 4:
