Here is an MVS/COBOL subroutine that if passed a DDname in the linkage
section (See: L01-COMM-AREA), it will return the DSN that is allocated
to that DDname.

       CBL NODYNAM,NOOPT,NOSSRANGE
       TITLE 'RETRIEVE JOB/STEP NAME PLUS DATASET INFO.'
       IDENTIFICATION DIVISION.

       PROGRAM-ID.  DPU020.
      *---------------------------------------------------------------*
      *
      *  USING POINTERS, RETRIEVE OPERATING-SYSTEM INFORMATION AND
      *  RETURN IT TO THE CALLER.
      *
      *---------------------------------------------------------------*
           EJECT
       ENVIRONMENT DIVISION.
       CONFIGURATION SECTION.

       INPUT-OUTPUT SECTION.
       FILE-CONTROL.

       DATA DIVISION.
       FILE SECTION.

      *---------------------------------------------------------------*
      * BEGINNING OF WORKING STORAGE
      *---------------------------------------------------------------*
       WORKING-STORAGE SECTION.

      *---------------------------------------------------------------*
      * W05-SUBSCRIPTS-POINTERS
      *---------------------------------------------------------------*

       01  FILLER PIC X(016) VALUE 'W05-SUBS-PNTRS**'.

       01  W05-SUBSCRIPTS-POINTERS.
           02  W05-TCB-ADDR-POINTER-X.
               03  W05-TCB-ADDR-POINTER       POINTER.

           02  W05-TIOT-SEG-POINTER           POINTER.
           02  W05-TIOT-BINARY  REDEFINES  W05-TIOT-SEG-POINTER
                                              PIC S9(09)  COMP.

           02  W05-JFCB-POINTER               POINTER.
           02  W05-JFCB-X       REDEFINES  W05-JFCB-POINTER.
               03  FILLER                     PIC X(01).
               03  W05-JFCB-LOWER-3           PIC X(03).
           EJECT
     
      *---------------------------------------------------------------*
      * W09-MISCELLANEOUS
      *---------------------------------------------------------------*

       01  FILLER PIC X(016) VALUE 'W09-MISC********'.

       01  W09-MISCELLANEOUS.
           02  W09-BINARY-2                PIC S9(04) COMP VALUE ZERO.
           02  FILLER  REDEFINES W09-BINARY-2.
               03  W09-BINARY-2L           PIC X(01).
               03  W09-BINARY-2R           PIC X(01).

           EJECT
      *---------------------------------------------------------------*
      *  LINKAGE AREA
      *---------------------------------------------------------------*
       LINKAGE SECTION.

       01  L01-COMM-AREA.
           02  L01-JOB-NAME            PIC X(08).
           02  L01-STEP-NAME           PIC X(08).
           02  L01-DD-NAME             PIC X(08).
           02  L01-DS-NAME             PIC X(44).
           02  L01-GDG-OR-MEMBER       PIC X(08).
           02  L01-CATALOG-IND         PIC X(01).
               88  L01-FILE-IS-CATALOGED   VALUE 'Y'.
           02  L01-FILE-TYPE           PIC X(01).
               88  L01-FILE-IS-GDG         VALUE 'G'.
               88  L01-FILE-IS-JES         VALUE 'J'.
               88  L01-FILE-IS-PDS         VALUE 'P'.
               88  L01-FILE-IS-REGULAR     VALUE 'R'.
           02  L01-VOLUME-COUNT        PIC 9(01).
           02  L01-VOLUMES             PIC X(30).

       01  L02-TCB-POINTER                 POINTER.

       01  L03-TCB.
           02  FILLER                      PIC X(012).
           02  L03-TIOT-POINTER            POINTER.

       01  L04-TIOT.
           02  L04-JOB-NAME                PIC X(08).
           02  L04-STEP-NAME               PIC X(08).

       01  L04-TIOT-START                  PIC X(24).

       01  L04-TIOT-SEGMENT.
           02  L04-TIOT-LENGTH             PIC X(01).
           02  FILLER                      PIC X(03).
           02  L04-DD-NAME                 PIC X(08).
           02  L04-JFCB-ADDRESS            PIC X(03).

       01  L05-JFCB.
           02  FILLER                      PIC X(16).
           02  L05-DS-NAME                 PIC X(44).
           02  L05-GDG-OR-MEMBER           PIC X(08).
           02  L05-DMI-BYTE                PIC X(01).
           02  FILLER                      PIC X(33).
           02  L05-DSTYPE-BYTE             PIC X(01).
           02  FILLER                      PIC X(30).
           02  L05-VOLUME-COUNT-BYTE       PIC X(01).
           02  L05-VOLUME-LIST             PIC X(30).
           EJECT
       PROCEDURE DIVISION USING L01-COMM-AREA.

           MOVE X'0000021C' TO W05-TCB-ADDR-POINTER-X
           SET ADDRESS OF L02-TCB-POINTER  TO W05-TCB-ADDR-POINTER.
           SET ADDRESS OF L03-TCB          TO L02-TCB-POINTER.
           SET ADDRESS OF L04-TIOT         TO L03-TIOT-POINTER.

           MOVE L04-JOB-NAME               TO L01-JOB-NAME.
           MOVE L04-STEP-NAME              TO L01-STEP-NAME.
           INITIALIZE L01-DS-NAME   L01-GDG-OR-MEMBER  L01-CATALOG-IND
                      L01-FILE-TYPE L01-VOLUME-COUNT   L01-VOLUMES.

      ****  IF THEY DID NOT SUPPLY A DD NAME, EXIT STAGE LEFT.

           IF L01-DD-NAME < 'A'
              GOBACK
           END-IF.

           MOVE LOW-VALUES TO W05-JFCB-X.
           SET ADDRESS OF L04-TIOT-START   TO L03-TIOT-POINTER.
           SET W05-TIOT-SEG-POINTER        TO L03-TIOT-POINTER.
           ADD 24                          TO W05-TIOT-BINARY.
           SET ADDRESS OF L04-TIOT-SEGMENT TO W05-TIOT-SEG-POINTER.

      ****  LOOP THROUGH ALL POINTERS TO THE DD NAMES.
      ****  WHEN WE GET A MATCH, MOVE IN ALL THE FILE DATA AND EXIT.

           PERFORM UNTIL L04-TIOT-LENGTH = LOW-VALUES
              MOVE L04-JFCB-ADDRESS           TO W05-JFCB-LOWER-3
              SET ADDRESS OF L05-JFCB         TO W05-JFCB-POINTER
              IF L04-DD-NAME = L01-DD-NAME
                 MOVE L04-DD-NAME             TO L01-DD-NAME
                 MOVE L05-DS-NAME             TO L01-DS-NAME
                 MOVE L05-GDG-OR-MEMBER       TO L01-GDG-OR-MEMBER
                 IF L05-DMI-BYTE > X'79' AND < X'90'
                    MOVE 'Y'                  TO L01-CATALOG-IND
                 ELSE
                    MOVE 'N'                  TO L01-CATALOG-IND
                 END-IF
                 IF L05-DMI-BYTE > X'19' AND < X'30'
                    MOVE 'J'                  TO L01-FILE-TYPE
                 ELSE
                    IF L05-DSTYPE-BYTE = X'00'
                       MOVE 'R'               TO L01-FILE-TYPE
                    ELSE
                       IF L05-DSTYPE-BYTE = X'01'
                          MOVE 'P'            TO L01-FILE-TYPE
                       ELSE
                          IF L05-DSTYPE-BYTE = X'02'
                             MOVE 'G'         TO L01-FILE-TYPE
                          ELSE
                             MOVE '?'         TO L01-FILE-TYPE
                          END-IF
                       END-IF
                    END-IF
                 END-IF
                 MOVE L05-VOLUME-COUNT-BYTE   TO W09-BINARY-2R
                 MOVE W09-BINARY-2            TO L01-VOLUME-COUNT
                 MOVE L05-VOLUME-LIST         TO L01-VOLUMES
                 GOBACK
              ELSE
                 ADD 20                    TO W05-TIOT-BINARY
                 SET ADDRESS OF L04-TIOT-SEGMENT
                                           TO W05-TIOT-SEG-POINTER
              END-IF
           END-PERFORM.

           GOBACK.
      *------------ END OF SUBROUTINE DPU020 -------------------------*

