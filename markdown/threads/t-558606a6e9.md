Dear COBOL-users!

I'd like to import a COBOL-file into MS Access2.0. The COBOL-file (which
gets downloaded from the host into an ansi-file) contains 8byte long
comp3-data. The attached Access2.0 function 'BIN2DBL' is supposed to
work with comp-data. How do I have to modify the algorithm to decompress
comp3??

I would appreciate your 'emailed' help very much.
Tin 

Function Bin2Dbl (ByVal b) As Double
    Dim negative As Integer: negative = (CInt(Asc(Mid$(b, 1, 1))) And
128)
    Dim i
    Dim p As Double: p = 1
    Dim x As Double: x = 0
    For i = 8 To 1 Step -1
        x = x + p * IIf(negative, 255 Xor CInt(Asc(Mid$(b, i, 1))),
CInt(Asc(Mid$(b, i, 1))))
        p = p * 256#
    Next i
    Bin2Dbl = IIf(negative, -1#, 1#) * x / 100#
End Function

I have not written one for Access, but here are a couple of BCD to decimal
routines I wrote in QBasic/QuickBasic.  The first converts an arbitrarily
long BCD string to a string of decimal digits, which can be converted to
any numeric type with VAL().  The second converts the string directly to
a LONG.  The logic should be very little, if any, different in Access.

'  *                                                *
'  *                 BCD2DEC.BAS                    *
'  *                                                *
'  *         Convert BCD string to Decimal          *
'  *                                                *
'  *              Judson D. McClendon               *
'  *              Sun Valley Systems                *
'  *              329 37th Court N.E.               *
'  *              Birmingham, AL 35215              *
'  *                 205-853-8440                   *
'  *                                                *
'  **************************************************

DECLARE FUNCTION Bcd2Dec$ (Bcd AS STRING)
DECLARE FUNCTION Dec2Bcd$ (Dec AS STRING)

DIM Dec AS STRING, Bcd AS STRING, I AS INTEGER
CLS

Dec = "00010203040506070809101112131415161718192021222324"
Dec = Dec + "25262728293031323334353637383940414243444546474849"
Dec = Dec + "50515253545556575859606162636465666768697071727374"
Dec = Dec + "75767778798081828384858687888990919293949596979899"
PRINT Dec

Bcd = Dec2Bcd$(Dec)

Dec = Bcd2Dec$(Bcd)

FUNCTION Bcd2Dec$ (Bcd AS STRING)
   CONST ASCII0 = 48     ' ASCII Zero = ASC("0")
   DIM I AS INTEGER, Dec AS STRING, D AS INTEGER

   Dec = ""
   FOR I = 1 TO LEN(Bcd)
      D = ASC(MID$(Bcd, I, 1))
      Dec = Dec + CHR$((D \ 16) + ASCII0) + CHR$((D AND 15) + ASCII0)
   NEXT
   Bcd2Dec$ = Dec
END FUNCTION

FUNCTION Dec2Bcd$ (Dec AS STRING)
   CONST ASCII0 = 48     ' ASCII Zero = ASC("0")
   DIM I AS INTEGER, D1 AS INTEGER, D2 AS INTEGER, Bcd AS STRING

   IF (LEN(Dec) MOD 2 = 1) THEN
      Dec = "0" + Dec
   END IF
   Bcd = ""
   FOR I = 1 TO LEN(Dec) STEP 2
      D1 = ASC(MID$(Dec, I, 1)) - ASCII0
      D2 = ASC(MID$(Dec, I + 1, 1)) - ASCII0
      Bcd = Bcd + CHR$(D1 * 16 + D2)
   NEXT
   Dec2Bcd$ = Bcd
END FUNCTION

DIM DEC AS LONG, BCD AS STRING
CLS
DEC = 1234567890
PRINT DEC
BCD = DEC2BCD$(DEC)
DEC = BCD2DEC&(BCD)
PRINT DEC

FUNCTION BCD2DEC& (BCD AS STRING)
   DIM I AS INTEGER, C AS STRING * 1, D AS INTEGER, DEC AS LONG
   DEC = 0
   FOR I = 1 TO LEN(BCD)
      DEC = DEC * 100
      D = ASC(MID$(BCD, I, 1))
      DEC = DEC + (D AND &HF) + (D \ 16) * 10
   NEXT
   BCD2DEC& = DEC
END FUNCTION

FUNCTION DEC2BCD$ (BYVAL DEC AS LONG)
   DIM C AS STRING * 1, D1 AS INTEGER, D2 AS INTEGER, BCD AS STRING
   BCD = ""
   DO WHILE DEC
      D2 = DEC MOD 10
      DEC = DEC \ 10
      D1 = DEC MOD 10
      DEC = DEC \ 10
      BCD = CHR$(D1 * 16 + D2) + BCD
   LOOP
   DEC2BCD$ = BCD
END FUNCTION
