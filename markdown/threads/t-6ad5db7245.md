Gene, here is what I use.



First cobol85.cbi    <====================================

[Compile Options]
ALPHAL=No
DLOAD=Yes
OPTIMIZE=Yes
TEST=No
LIB=.;%ETK32DIR%
MAIN=No
RSV=ALL
SRF=FIX,FIX
TAB=8
[COBOL85]
Language=E


then cobol.bat  for making DLL  <=================================:

@echo off
::   COBOL - compilation using FUJITSU COBOL
::   Date-Written:  97/03/09    Leif Svalgaard
::       -Revised:  97/09/22    Leif Svalgaard

if exist quiet.q$q   del quiet.q$q   > nul
if exist cobol85.reg del cobol85.reg > nul

if '%1' == '/Q' goto quiet
if '%1' == '/q' goto quiet
goto continue

:quiet
echo quiet>quiet.q$q
shift

:continue
if '%1' == '' goto help

:repeat
if exist %1.err del %1.err > nul
if exist %1.COB goto exists

:error
%lib%\cobol32 -i %etkdir%\cobol85.cbi %1.cob > %1.err

:stop
type %1.err | more
pause
goto end

:show
type %1.err | more
pause
goto loop

:try_again
:exists
if exist %1.obj del %1.obj > nul
if exist quiet.q$q goto compile

echo Compiling  %1...
:compile
%lib%\cobol32 -i %etkdir%\cobol85.cbi %1.cob > nul

:test
if errorlevel 1     goto error
if not exist %1.obj goto error
call linkup %1

:loop
shift

it uses linkup.bat <=====================================:

@echo off
::   LINKUP - link a program to DLL
::
::   Date-Written:  97/03/09    Leif Svalgaard
::       -Revised:  97/09/22    Leif Svalgaard
::
if '%1' == '' goto help

:loop
if '%1' == '' goto end

if exist %1.obj goto link
echo %1.obj not found
goto end

:link
rem echo making DLL %1...
echo LIBRARY %1 >  %1.exp
echo EXPORTS %1 >> %1.exp
upper<%1.exp>%1.def

%lib%\lib/nologo /def:%1.def /out:%1.lib /machine:ix86 %1.obj  > nul
%lib%\link/nologo %1.obj %1.exp /dll /out:%1.dll @%etkdir%\f3libs

:done
if exist %1.def  del  %1.def > nul
if exist %1.exp  del  %1.exp > nul
if exist %1.lib  del  %1.lib > nul
if exist %1.obj  del  %1.obj > nul
shift
goto loop

:help
echo ����������������������������������������������������ͻ

:end
