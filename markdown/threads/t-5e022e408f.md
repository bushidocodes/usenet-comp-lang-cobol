simply executing the Cobol program using the command line interpreter from
inside Basic, but that still allows you to do what you want.  You will have
to modify the Cobol programs to accept the file name from the command line,
and then insert the filename into the literal. For example, the one below
would be modified to 'ShellHandle = Shell("RE01.EXE MYFILE.DATA",
vbNormalFocus', and the Cobol program modified to pick up "MYFILE.DATA" as a
command line argument.

Or you just use:
      *-----------------------------------------------------------------
       LINKAGE Section.
      *-----------------------------------------------------------------
       01 CallParam.
           03 CallParamLength  pic 9(4) comp-x.
           03 CallParamLine.
               05 CallParamChar pic x occurs 1 to 999
                                      depending on CallParamLength.

      *-----------------------------------------------------------------
       PROCEDURE DIVISION using CallParam.

The params will be stored in the structure above and you can parse them
very easily doing soemthing like:

            Move 1 to naechst-argv
            perform until naechst-argv > CallParamLength
               add 1 to dIdx
               unstring CallParamLine delimited by space
                   into argv
                   count in argv-laenge
                   with pointer naechst-argv
               end-unstring
               if Argv-Laenge > Argv-Max-Laenge
                *> some error processing
               else
                *> process the datas
               end-if
            end-perform

It works with Merant Cobol but I don't know if it works with other
compilers as well.

HTH - Stefan Meyer

Here�s a brief explanation on the following code.

The only �got ya� you really have to watch for is the way windows looks

Option Explicit

Private Declare Function ShellExecute Lib "shell32.dll" Alias _
"ShellExecuteA" (ByVal hwnd As Long, ByVal lpszOp As _
String, ByVal lpszFile As String, ByVal lpszParams As String, _
ByVal lpszDir As String, ByVal FsShowCmd As Long) As Long

Private Declare Function GetDesktopWindow Lib "user32" () As Long

Const SW_SHOWNORMAL = 1

Const SE_ERR_FNF As String = 2
Const SE_ERR_PNF As String = 3
Const SE_ERR_ACCESSDENIED As String = 5
Const SE_ERR_OOM As String = 8
Const SE_ERR_DLLNOTFOUND As String = 32
Const SE_ERR_SHARE As String = 26
Const SE_ERR_ASSOCINCOMPLETE As String = 27
Const SE_ERR_DDETIMEOUT As String = 28
Const SE_ERR_DDEFAIL As String = 29
Const SE_ERR_DDEBUSY As String = 30
Const SE_ERR_NOASSOC As String = 31
Const ERROR_BAD_FORMAT As String = 11

Function StartDoc(DocName As String, Parameters As String, DefaultDir
As String) As Long
    Dim Scr_hDC As Long
    Scr_hDC = GetDesktopWindow()
    StartDoc = ShellExecute(Scr_hDC, "Open", DocName, _
    Parameters, DefaultDir, SW_SHOWNORMAL)
End Function

Public Sub modcmdCallWithParms_Click()
    Dim r As Long, msg As String
    Dim strDocName As String
    Dim strParameters As String
    Dim strDefaultDir As String
    strDocName = "D:\eone4win\vers41\programs\ezlmo.exe"
    strParameters = "         B:Y C:Y D:Y"
    strDefaultDir = "d:\eone4win\vers41\programs"

    r = StartDoc(strDocName, strParameters, strDefaultDir)
    If r <= 32 Then
        'There was an error
        Select Case r
            Case SE_ERR_FNF
                msg = "File not found."
            Case SE_ERR_PNF
                msg = "Path not found."
            Case SE_ERR_ACCESSDENIED
                msg = "Access denied."
            Case SE_ERR_OOM
                msg = "Out of memory."
            Case SE_ERR_DLLNOTFOUND
                msg = "DLL not found."
            Case SE_ERR_SHARE
                msg = "A sharing violation occurred."
            Case SE_ERR_ASSOCINCOMPLETE
                msg = "Incomplete or invalid file association."
            Case SE_ERR_DDETIMEOUT
                msg = "DDE Time out."
            Case SE_ERR_DDEFAIL
                msg = "DDE transaction failed."
            Case SE_ERR_DDEBUSY
                msg = "DDE busy."
            Case SE_ERR_NOASSOC
                msg = "No association for file extension."
            Case ERROR_BAD_FORMAT
                msg = "Invalid EXE file or error in EXE image."
            Case Else
                msg = "Unknown error."
        End Select
        MsgBox msg & "  Contact CGW.", vbExclamation, "Error"
    End If
End Sub
