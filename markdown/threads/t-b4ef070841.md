By the way, the previous example is how you would do it in APL. Here's
a C version you might be able to translate into Cobol:

#include <stdio.h>

void fib(int n)
{
   int i;
   unsigned num1, num2, temp;

   if (n < 1)    // Error
   {
       printf("Error\n");
       return;
   }

   if (n == 1)
       printf("0\n");
   else if (n == 2)
       printf("0-1\n");
   else
   {
       num1 = 0;
       num2 = 1;
       printf("0-1");
       for (i = 2; i < n; ++i)
       {
           printf("-%d", num1 + num2);
           temp = num2;
           num2 = num2 + num1;
           num1 = temp;
       }
       printf("\n");
   }
}

int main(void)
{
   fib(25);
   return 0;
}
