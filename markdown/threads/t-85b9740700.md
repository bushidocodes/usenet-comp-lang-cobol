  FD STOCKMAST.
  01 STOCKM-REC.
  03 M-PART-NUM  PIC X(8).
   88 END-OF-MAST VALUE HIGH-VALUES.
  03 L-TRANS-DATE  PIC 9(6).
  03 BALANCE   PIC S9(6).
  03 MIN   PIC 9(6).
  03 MAX   PIC 9(6).

  FD QUERYFILE.
  01 QUERY-REC   PIC X(33).

  FD NEW-STOCKMAS.
  01 N-STOCK-REC   PIC X(33).

  FD PRINT-FILE
  LINAGE IS 60 LINES
  WITH FOOTING AT 56
  LINES AT TOP 2
  LINES AT BOTTOM 4.
  01 PRINT-REC   PIC X(68).

  WORKING-STORAGE SECTION.


  01 W-MAIN-HEAD.
         03    PIC X(34) VALUE SPACES.
         03    PIC X(12) VALUE
                 "ERROR REPORT".

  01 W-SUB-HEAD.
  03    PIC XXX VALUE SPACES.
  03    PIC X(12) VALUE
   "PART NO.".
  03    PIC X(13) VALUE
   "TRANS TYPE".
  03    PIC X(17) VALUE
   "CUST/SUPP NO.".
  03    PIC X(14) VALUE
   "TRANS DATE".
  03    PIC X(8) VALUE
   "QUANTITY".

  01 W-DETAIL.
  03    PIC XXX VALUE SPACES.
  03 P-PART-NUM  PIC X(8).
  03    PIC X(8) VALUE SPACES.
  03 P-TRANS-TYPE  PIC X.
  03    PIC X(11) VALUE SPACES.
  03 P-C-S-NUM  PIC Z(5)9.
  03    PIC X(10) VALUE SPACES.
  03 P-TRANS-DATE  PIC Z(5)9.
  03    PIC X(7) VALUE SPACES.
  03 P-QUANTITY  PIC Z(5)9.

  PROCEDURE DIVISION.
  MAIN-CONTROL.
  PERFORM INITIAL-PROCESS
  PERFORM UPDATE-PROCESS UNTIL
   END-OF-TRANS AND END-OF-MAST
  PERFORM FINAL-PROCESS
  STOP RUN.

  INITIAL-PROCESS.
  OPEN  INPUT  TRANSFILE
    STOCKMAST
   OUTPUT NEW-STOCKMAS
    QUERYFILE
    PRINT-FILE
  PERFORM READ-TRANSFILE
  PERFORM READ-STOCKMAST.
/  PERFORM NEW-HEADINGS.

  UPDATE-PROCESS.
   EVALUATE TRUE
    WHEN PART-NUM < M-PART-NUM
/     PERFORM PRINT-DETAIL
     PERFORM READ-TRANSFILE
    WHEN PART-NUM > M-PART-NUM
            WRITE N-STOCK-REC FROM STOCKM-REC
     PERFORM READ-STOCKMAST
    WHEN PART-NUM = M-PART-NUM
     PERFORM CALCULATE-DATA
     PERFORM READ-TRANSFILE
   END-EVALUATE.

  CALCULATE-DATA.
   MOVE TRANS-DATE TO L-TRANS-DATE
   IF TRANS-TYPE = "R" THEN
    COMPUTE BALANCE = BALANCE + QUANTITY
    ELSE IF TRANS-TYPE = "I" THEN
     COMPUTE BALANCE = BALANCE - QUANTITY
    END-IF
   END-IF
   IF BALANCE < MIN OR BALANCE > MAX THEN
    WRITE QUERY-REC FROM STOCKM-REC
   END-IF.

  READ-TRANSFILE.
   READ TRANSFILE AT END
    MOVE HIGH-VALUES TO PART-NUM
   END-READ.

  READ-STOCKMAST.
   READ STOCKMAST AT END
    MOVE HIGH-VALUES TO M-PART-NUM
   END-READ.

  NEW-HEADINGS.
          WRITE PRINT-REC FROM W-MAIN-HEAD AFTER PAGE
   WRITE PRINT-REC FROM W-SUB-HEAD AFTER 3.

  PRINT-DETAIL.
   MOVE TRANS-TYPE TO P-TRANS-TYPE
   MOVE PART-NUM TO P-PART-NUM
   MOVE C-S-NUM TO P-C-S-NUM
   MOVE TRANS-DATE TO P-TRANS-DATE
   MOVE QUANTITY TO P-QUANTITY
   WRITE PRINT-REC FROM W-DETAIL AFTER 1
    AT END-OF-PAGE
     PERFORM NEW-HEADINGS
   END-WRITE.

  FINAL-PROCESS.
   CLOSE TRANSFILE
    STOCKMAST
    NEW-STOCKMAS
    QUERYFILE
/    PRINT-FILE.

 Repost of source code in case you've lost it


IDENTIFICATION DIVISION.
 PROGRAM-ID. QUESTION-B.

 ENVIRONMENT DIVISION.
 INPUT-OUTPUT SECTION.
 FILE-CONTROL.
  SELECT TRANSFILE ASSIGN TO "TRANS.DAT"
   ORGANIZATION LINE SEQUENTIAL.
  SELECT STOCKMAST ASSIGN TO "STOCKMA.DAT"
   ORGANIZATION LINE SEQUENTIAL.
  SELECT NEW-STOCKMAS ASSIGN TO "NSTOCKMA.DAT"
   ORGANIZATION LINE SEQUENTIAL.
  SELECT QUERYFILE ASSIGN TO "QUERY.DAT"
   ORGANIZATION LINE SEQUENTIAL.
  SELECT PRINT-FILE ASSIGN TO PRINTER.

 DATA DIVISION.
 FILE SECTION.
 FD TRANSFILE.
 01 TRANS-REC.
         03 PART-NUM  PIC X(8).
   88 END-OF-TRANS VALUE HIGH-VALUES.
  03 TRANS-TYPE  PIC X.
         03 C-S-NUM   PIC 9(6).
  03 TRANS-DATE  PIC 9(6).
  03 QUANTITY  PIC 9(6).

 FD STOCKMAST.
 01 STOCKM-REC.
  03 M-PART-NUM  PIC X(8).
   88 END-OF-MAST VALUE HIGH-VALUES.
  03 L-TRANS-DATE  PIC 9(6).
  03 BALANCE   PIC S9(6).
  03 MIN   PIC 9(6).
  03 MAX   PIC 9(6).

 FD QUERYFILE.
 01 QUERY-REC   PIC X(33).

 FD NEW-STOCKMAS.
 01 N-STOCK-REC   PIC X(33).

 FD PRINT-FILE
  LINAGE IS 60 LINES
  WITH FOOTING AT 56
  LINES AT TOP 2
  LINES AT BOTTOM 4.
 01 PRINT-REC   PIC X(71).

 WORKING-STORAGE SECTION.


 01 W-MAIN-HEAD.
         03    PIC X(34) VALUE SPACES.
         03    PIC X(12) VALUE
                 "ERROR REPORT".

 01 W-SUB-HEAD.
  03    PIC X(5) VALUE SPACES.
  03    PIC X(12) VALUE
   "PART NO.".
  03    PIC X(13) VALUE
   "TRANS TYPE".
  03    PIC X(17) VALUE
   "CUST/SUPP NO.".
  03    PIC X(14) VALUE
   "TRANS DATE".
  03    PIC X(8) VALUE
   "QUANTITY".

 01 W-DETAIL.
  03    PIC X(5) VALUE SPACES.
  03 P-PART-NUM  PIC X(8).
  03    PIC X(8) VALUE SPACES.
  03 P-TRANS-TYPE  PIC X.
  03    PIC X(11) VALUE SPACES.
  03 P-C-S-NUM  PIC X(6).
  03    PIC X(10) VALUE SPACES.
  03 P-TRANS-DATE  PIC X(6).
  03    PIC X(7) VALUE SPACES.
  03 P-QUANTITY  PIC Z(6)9.

 PROCEDURE DIVISION.
 MAIN-CONTROL.
  PERFORM INITIAL-PROCESS
  PERFORM UPDATE-PROCESS UNTIL
   END-OF-TRANS AND END-OF-MAST
  PERFORM FINAL-PROCESS
  STOP RUN.

 INITIAL-PROCESS.
  OPEN  INPUT  TRANSFILE
    STOCKMAST
   OUTPUT NEW-STOCKMAS
    QUERYFILE
    PRINT-FILE
  PERFORM READ-TRANSFILE
  PERFORM READ-STOCKMAST.
  PERFORM NEW-HEADINGS.

  UPDATE-PROCESS.
   EVALUATE TRUE
    WHEN PART-NUM < M-PART-NUM
     PERFORM PRINT-DETAIL
     PERFORM READ-TRANSFILE
    WHEN PART-NUM > M-PART-NUM
            WRITE N-STOCK-REC FROM STOCKM-REC
     PERFORM READ-STOCKMAST
    WHEN PART-NUM = M-PART-NUM
     PERFORM CALCULATE-DATA
     PERFORM READ-TRANSFILE
     END-EVALUATE.

  CALCULATE-DATA.
   MOVE TRANS-DATE TO L-TRANS-DATE
   IF TRANS-TYPE = "R" THEN
    COMPUTE BALANCE = BALANCE + QUANTITY
   ELSE
    COMPUTE BALANCE = BALANCE - QUANTITY
   END-IF
   IF BALANCE < MIN OR BALANCE > MAX THEN
    WRITE QUERY-REC FROM STOCKM-REC
   END-IF.

  READ-TRANSFILE.
   READ TRANSFILE AT END
    MOVE HIGH-VALUES TO PART-NUM
   END-READ.

  READ-STOCKMAST.
   READ STOCKMAST AT END
    MOVE HIGH-VALUES TO M-PART-NUM
   END-READ.

  NEW-HEADINGS.
          WRITE PRINT-REC FROM W-MAIN-HEAD
   WRITE PRINT-REC FROM W-SUB-HEAD AFTER 3.

  PRINT-DETAIL.
   MOVE TRANS-TYPE TO P-TRANS-TYPE
   MOVE PART-NUM TO P-PART-NUM
   MOVE C-S-NUM TO P-C-S-NUM
   MOVE TRANS-DATE TO P-TRANS-DATE
   MOVE QUANTITY TO P-QUANTITY
   WRITE PRINT-REC FROM W-DETAIL AFTER 1
    AT END-OF-PAGE
     PERFORM NEW-HEADINGS
   END-WRITE.

  FINAL-PROCESS.
   CLOSE TRANSFILE
    STOCKMAST
    NEW-STOCKMAS
    QUERYFILE
    PRINT-FILE
