DeclareCursor: procedure expose global. LocalGlobal.
/***************************************************************************
SQLstatment passed.
Declare cursor for SQLSatatment.
Prepare SQLStatement.
Open Cursor.
***************************************************************************/
data. = ""
SQLStatement = arg(1)
Call SQLEXEC 'DECLARE c1 CURSOR FOR s1'
if result <> 0 | SQLCA.SQLCODE <> 0 Then
   signal db2ERROR
sqlda = ""
Call SQLEXEC 'PREPARE s1 FROM :SQLStatement'

if result <> 0 | SQLCA.SQLCODE <> 0 Then Do
   signal db2ERROR
End

Call SQLEXEC 'DESCRIBE s1 INTO :Global.Columns '
if result <> 0 | SQLCA.SQLCODE <> 0 Then do
   signal db2ERROR
end

call SQLEXEC 'OPEN c1'
if result <> 0 | SQLCA.SQLCODE <> 0 Then
   signal DB2ERROR

return SQLCA.SQLCODE


Fetch_Statement: procedure expose global. data. LocalGlobal.
/***************************************************************************
Fetch data.
If no more rows
   Close cursor
else if Row found
   move value to column name.
otherwise
   There is a problem. Kill program.
***************************************************************************/
if arg(1) = "Expose" then
   return 0

call SQLEXEC 'FETCH c1 USING DESCRIPTOR :sqlda'
select
   when SQLCA.SQLCODE = 100 then do
      call SQLEXEC 'CLOSE c1'
      if result <> 0 | SQLCA.SQLCODE <> 0 Then
         signal DB2ERROR
      return 100
   end
   when SQLCA.SQLCODE = 0 then do
      do i=1 to global.Columns.sqld
/* SQLTypes          VCHAR                   CHAR
                    LVCHAR                VCHAR w/NULL
                  CHAR w/NULL             LVCHAR w/NULL
                     Date                    Time
                  Timestamp               Date w/NULL
                  Time w/NULL             Timestamp w/NULL
*/
         if sqlda.i.sqltype = 448 | sqlda.i.sqltype = 452 |,
            sqlda.i.sqltype = 456 | sqlda.i.sqltype = 449 |,
            sqlda.i.sqltype = 453 | sqlda.i.sqltype = 457 |,
            sqlda.i.sqltype = 384 | sqlda.i.sqltype = 388 |,
            sqlda.i.sqltype = 392 | sqlda.i.sqltype = 385 |,
            sqlda.i.sqltype = 389 | sqlda.i.sqltype = 393 then do

            if sqlda.i.sqlind = -1 then
               sqlda.i.sqldata = ""
            start_pos = pos('"',sqlda.i.sqldata)
            do while (Start_Pos) > 0
               sqlda.i.sqldata = insert('"',sqlda.i.sqldata,start_pos)
               start_pos = pos('"',sqlda.i.sqldata,start_pos+2)
            end
            Assign_statment = 'Data.'strip(Global.Columns.i.sqlname)' = "',
              ||strip(sqlda.i.sqldata)'"'
         end
         else do
            if sqlda.i.sqlind = -1 then
               sqlda.i.sqldata = 0
            Assign_statment = "Data."Global.Columns.i.sqlname" = ",
               sqlda.i.sqldata
         end
         interpret assign_statment
      end
   end
   otherwise
      signal DB2ERROR
end
return SQLCA.SQLCODE

Parse_ARGS:
/****************************************************************
** If no arguments or Help switches.
**    Goto Help Display
** Parse various switches until no more switches.
** --While switches exist...
** --   Find next switch
** --   Parse Switch & Value to seperate them out
** --   (Note: Requires that a colon exist as part of the switch)
** --   For switch do appropriate action
*****************************************************************/

If ARG(1) = '' | POS('/?',ARG(1)) | POS('/H',TRANSLATE(ARG(1))) THEN
   SIGNAL Usage

SwitchPos = pos('/',ARG(1))
Do While SwitchPos > 0
   EndOfSwitchInd = pos(' ',arg(1),SwitchPos)
   if EndOfSwitchInd < 1 then
      EndSwitchPos = Length(arg(1))
   else do
      EndSwitchPos = pos(' ',Arg(1),EndOfSwitchInd) - 1
      if EndSwitchPos < 1 then
         EndSwitchPos = Length(arg(1))
   end

   Switch_value = substr(arg(1),SwitchPos,EndSwitchPos-SwitchPos+1)

   parse value Switch_Value with '/' Switch ':' Value
   Switch = TRANSLATE(Switch)
   SwitchInd = Substr(Switch,1,1)
   ValueInd = Translate(Substr(Value,1,1))
   Select
      /* Are Comments Requested? */
      when SwitchInd = 'C' then do
         If ValueInd = 'Y' | ValueInd = 'T' |,
            ValueInd = '1' | ValueInd = '' then
            CommentsRequested = 1
         else
            CommentsRequested = 0
      end  /* Do */

      /* Are Redefines Requested? */
      when SwitchInd = 'R' then do
         If ValueInd = 'Y' | ValueInd = 'T' |,
            ValueInd = '1' | ValueInd = '' then
            RedefinesRequested = 1
         else
            RedefinesRequested = 0
      end  /* Do */

      /* Is Display Requested? */
      when Switch = 'DISPLAY' then do
         If ValueInd = 'Y' | ValueInd = 'T' |,
            ValueInd = '1' | ValueInd = '' then
            DisplayRequested = 1
         else
            DisplayRequested = 0
      end  /* Do */

      /* Are NULL Indicators Requested? */
      when SwitchInd = 'N' then do
         If ValueInd = 'Y' | ValueInd = 'T' |,
            ValueInd = '1' | ValueInd = '' then
            NullIndicatorsRequested = 1
         else
            NullIndicatorsRequested = 0
      end  /* Do */

      /* Is EXEC SQL Requested? */
      when SwitchInd = 'E' then do
         If ValueInd = 'Y' | ValueInd = 'T' |,
            ValueInd = '1' | ValueInd = '' then
            EXECSQLRequested = 1
         else
            EXECSQLRequested = 0
      end  /* Do */

      /* Database mentioned? */
      when SwitchInd = 'D' & Switch <> 'DISPLAY' then do
         DatabaseName = Value
      end  /* Do */

      /* Table Mentioned? */
      when SwitchInd = 'T' then do
         TableName = Value
      end  /* Do */
