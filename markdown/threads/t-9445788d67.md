      DECLARE Origin FIXED BINARY (31);
      DECLARE 1 somedate,
                  2 d CHARACTER (2),
                  2 m CHARACTER (2),
                  2 y CHARACTER (4);
      DECLARE J FIXED BINARY (31);

      Origin = DAYS ('30121923', 'DDMMYYYY');
      DO J = DAYS ('30121923', 'DDMMYYYY') DAYS ('31122100', 'DDMMYYYY');
         STRING(somedate) = DAYSTODATE (J, 'DDMMYYYY');
         IF dateval(m, d, y)+origin ^= J THEN
            PUT SKIP EDIT (somedate, dateval (m, d, y)+origin, J) (A, X(1));
      END;
   END TRY_DATE;
