and find that the likely cause is eliminated and the state of the
tests has totally shifted the percentages of the next test taking into
account the tests that have already been performed.  i.e.:  The next

       5mm-full-plus-half-repo-1.

           
           *> reposition procedure for channels that need
           *> to be repostioned full + %<whatever> times, based on

          Move ws-5-round to g90-valuet
          perform varying Loop-CNT from 0 by 1 until Loop-cnt = 2
           if loop-CNT = 0
              move ws-chanhole-width to g90-valuey
                                        xyvaluey
           else
              move ws-outside-hole-dim to g90-valuey
                                          xyvaluey
           end-if

           move ws-initial5hole to g90-valuex
           move ws-5-round to g90-valuet
           invoke self "ComputeLnCNT" using  g90-n-cnt g90
           move ws-g90 to outputtap-record
             write outputtap-record
           move zeros to g90-valuex, g90-valuey, g90-valuet
                                      *>400   = 250
           if Loop-CNT = 0
              subtract ws-both-45 from ws-chan-num-bend-a giving
              ws-punchable-a
           end-if
           if ws-punchable-a > ws-punchable-area
           *>first bend is greater than the origin will allow.
              Subtract ws-both-45 from ws-punchable-area
              giving ws-punchable-area2
              Subtract ws-punchable-area2 from ws-punchable-a
              giving ws-first-a
              if ws-first-a > ws-originxvalue
                 Add ws-chanpitchvalue to ws-both-45 giving
                 g90-valuex
                 if Loop-CNT = 0
                    Subtract ws-chanpitchvalue from ws-first-a
                 end-if
                 Invoke self "ComputeLnCNT" using g90-n-cnt g90
                 Move ws-g90 to outputtap-record
                  write outputtap-record

                 *>test again to seee if we get more holes.
                 if ws-first-a > ws-chanpitchvalue
                 Compute ws-calc-k =
                         ( ws-first-a / ws-chanpitchvalue )
                 Move ws-calc-k to g282-valuek
                 Move 0.00 to g282-valuej
                 Move ws-chanpitchvalue to g282-valuei
                 Invoke self "ComputeLnCNT" using g282-n-cnt g282
                 Move ws-g282 to outputtap-record
                  write outputtap-record
                  Move zeros to g282-valuei,
                                g282-valuek,
                 end-if
                 if Loop-CNT = 0
                    Subtract ws-first-a from ws-chan-num-bend-a
                 end-if
              end-if
              else

           if ws-punchable-a > ws-chanpitchvalue


              add ws-chanpitchvalue to ws-both-45 giving xyvaluex
              Invoke self "ComputeLnCNT" using xy-n-cnt xy
              Move ws-xy to outputtap-record
               write outputtap-record
               Move zeros to xyvaluex
              if Loop-CNT = 0
                subtract ws-chanpitchvalue from ws-punchable-a
              end-if
              compute ws-sum =
                      ( ws-punchable-a + ws-chanpitchvalue ) +
                      ( ws-both-45 )
              if ws-sum > ws-originxvalue
                 if Loop-CNT = 0
                     Subtract ws-originxvalue from ws-punchable-a
                     giving ws-punchable-a2 *> <-means first origin area
                     subtract ws-punchable-a2 from ws-punchable-a
                 end-if
                 compute ws-calc-k =
                         ( ws-punchable-a2 / ws-chanpitchvalue )
                         move ws-calc-k to g282-valuek
                         move ws-chanpitchvalue to g282-valuei
                         move 0.00 to g282-valuej
                         invoke self "ComputeLnCNT" using  g282-n-cnt
                                                           g282
                         move ws-g282 to outputtap-record
                          write outputtap-record


              else
                *>then the bend value is less than originX, so we
                *>can put all holes in this end
                 compute ws-calc-k =
                         ( ws-punchable-a / ws-chanpitchvalue )
                         move ws-calc-k to g282-valuek
                         move ws-chanpitchvalue to g282-valuei
                         move 0.00 to g282-valuej
                         invoke self "ComputeLnCNT" using  g282-n-cnt
                                                           g282
                         move ws-g282 to outputtap-record
                          write outputtap-record
                         Move zeros to ws-repo-pos-A
                         if Loop-CNT = 0
                            subtract ws-chan-num-bend-a from
                                     ws-chan-num-length giving
                                     ws-chan-num-bend-b
                            subtract ws-both-45 from ws-chan-num-bend-b
                         end-if
                         Move zeros to ws-calc-k
                         compute ws-calc-k =
                             ( ws-chan-num-bend-b / ws-chanpitchvalue )
                         multiply ws-calc-k by ws-chanpitchvalue
                         giving ws-sum
                         if Loop-CNT = 0
                            compute ws-sum2 =
                                    ( ws-sum + ws-both-45 )
                            subtract ws-sum2 from ws-chan-num-length
                            giving ws-first-b-hole
                         end-if

      *--------------->this is to signal that we have finished
      *--------------->processing bend 'A'
                         Move zeros to ws-chan-num-bend-A

                         if ws-first-b-hole <= ws-punchable-range
                            move ws-first-b-hole to xvaluex
                            invoke self "ComputeLnCNT" using x-n-cnt
                                                             xx
                            move ws-x to outputtap-record
                             write outputtap-record
                             move zeros to xvaluex

                            Move zeros to ws-left
                            subtract ws-first-b-hole from
                            ws-punchable-range giving ws-left

                            *>we test to see if we can get more holes
                            *>in the channel from this origin.
                            if ws-left > ws-chanpitchvalue
                               add ws-chanpitchvalue to ws-first-b-hole
                               giving xvaluex
                               invoke self "ComputeLnCNT" using x-n-cnt
                                                                xx
                               move ws-x to outputtap-record
                                write outputtap-record
                               move zeros to xvaluex
                               move zeros to ws-calc-k
                               subtract ws-chanpitchvalue from ws-left
                               if ws-left > ws-chanpitchvalue
                                  compute ws-calc-k =
                                       ( ws-left / ws-chanpitchvalue )
                                       move ws-calc-k to g282-valuek
                                       move 0.00 to g282-valuej
                                       move ws-chanpitchvalue to
                                                            g282-valuei
                               end-if
                               *>calculate what punchable area is left
                               *>in the "B" section.
                               subtract ws-first-b-hole from
                               ws-chan-num-length
                               giving ws-repo-pos-B
                               subtract ws-both-45 from ws-repo-pos-B
                               if ws-calc-k <> zero
                                  compute ws-sum =
                                    ( ws-calc-k * ws-chanpitchvalue ) +
                                    ws-chanpitchvalue
                                  subtract ws-sum from ws-repo-pos-B
                               end-if
                            else
                               *>then only one extra hole is processed
                               *>in the B dimension.
                               subtract ws-first-b-hole from
                                         ws-chan-num-length giving
                                         ws-repo-pos-B
                               subtract ws-both-45 from ws-repo-pos-B
      *                        move 1 to Loop-CNT
                            end-if
                         else
      *                     Then the first hole in the extra flange
      *                     is greater than the originX value.

      *                     if Loop-CNT = 0
      *                       subtract ws-chan-num-bend-a from
      *                       ws-chan-num-length
      *                                  giving ws-chan-num-bend-b
      *                       move ws-chan-num-bend-b to ws-chan-left
      *                       subtract ws-both-45 from ws-chan-left
      *                     end-if
                            continue

                         end-if
           else
      *    No extra 5.0mm holes can be processed in the "A" dim.

           *>work out all holes backwards here:
           if Loop-CNT = 0
              subtract ws-both-45 from ws-chan-num-bend-b
           end-if
           compute ws-calc-k =
               ( ws-chan-num-bend-b / ws-chanpitchvalue )
           if Loop-CNT = 0
              multiply ws-calc-k by ws-chanpitchvalue
              giving ws-sum
              compute ws-sum2 =
                      ( ws-sum + ws-both-45 )
              subtract ws-sum2 from ws-chan-num-length giving
              ws-first-b-hole
           end-if
           if ws-first-b-hole < ws-punchable-range
              move ws-first-b-hole to xvaluex
              invoke self "ComputeLnCNT" using x-n-cnt
                                               xx
              move ws-x to outputtap-record
               write outputtap-record
               move zeros to xvaluex

              subtract ws-first-b-hole from
              ws-punchable-range giving ws-left

              *>we test to see if we can get more holes
              *>in the channel from this origin.
              if ws-left > ws-chanpitchvalue
                 add ws-chanpitchvalue to ws-first-b-hole
                 giving xvaluex
                 invoke self "ComputeLnCNT" using x-n-cnt
                                                  xx
                 move ws-x to outputtap-record
                  write outputtap-record
                 move zeros to xvaluex
                 move zeros to ws-calc-k
                 subtract ws-chanpitchvalue from ws-left
                 if ws-left > ws-chanpitchvalue
                    compute ws-calc-k =
                         ( ws-left / ws-chanpitchvalue )
                         move ws-calc-k to g282-valuek
                         move 0.00 to g282-valuej
                         move ws-chanpitchvalue to
                                              g282-valuei
                 end-if
                 *>calculate what punchable area is left
                 *>in the "B" section.
                 subtract ws-first-b-hole from
                 ws-chan-num-length giving ws-chan-left
                 subtract ws-both-45 from ws-chan-left
                 if ws-calc-k <> zero
                    compute ws-sum =
                      ( ws-calc-k * ws-chanpitchvalue ) +
                      ws-chanpitchvalue
                    subtract ws-sum from ws-chan-left
                 end-if
      *          Move 1 to Loop-CNT

              else
                 *>then only one extra hole is processed
                 *>in the B dimension.
                 subtract ws-first-b-hole from
                           ws-chan-num-length giving
                           ws-chan-left
                 subtract ws-both-45 from ws-chan-left
              end-if
           else
      *       Then the first hole in the extra flange
      *       is greater than the originX value.
             if Loop-CNT = 0
              subtract ws-chan-num-bend-a from
              ws-chan-num-length
                         giving ws-chan-num-bend-b
              move ws-chan-num-bend-b to ws-chan-left
              subtract ws-both-45 from ws-chan-left
             end-if
           end-if
           end-perform

           *>set counter back to zero
           move 0 to Loop-CNT
