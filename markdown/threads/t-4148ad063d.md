Sure.  For DOS, though, you will need to use assembler.  For windows, I use a purchased DLL.  The routine below was written for MF Cobol version 4.0 running under DOS.  Sorry it looks so shitty placed into a message, but the tabs do not line up the same. The source is actually quite nice looking (would not want anybody to think I wrote hard to follow code)<G>

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 TITLE 'SERINT---'A serial interupt/scale routine for COBOL' ;
 PAGE ,120       ;
;         ;
;     COPYRIGHT (C) 1986                            ;
;      Tees & Tees Inc.    ;
;         Written by    ;
;         D. L. Tees    ;
;        VERSION 1.1    ;
;         ;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
 INCLUDE \LIBRARY\MACROS\SETUPIO.MAC
 INCLUDE \LIBRARY\MACROS\CODES.MAC
 EXTRN GET_SCREEN_BASE:NEAR
;
CSEG SEGMENT PARA PUBLIC 'CODE'
 ASSUME CS:CSEG,DS:CSEG,ES:CSEG
DYNS STRUC
 DW ?   ;Pushed BP
 DD ?   ;Long address
PARM1 DW ?   ;Pointer to arguement 1
BASE1 DW ?   ;segment for arguement 1
PARM2 DW ?   ;Pointer to arguement 2
BASE2 DW ?   ;segment for arguement 2
PARM3 DW ?   ;Pointer to arguement 3
BASE3 DW ?   ;segment for arguement 3
DYNS ENDS
;
;
SCREEN_OFFSET DW 0
SCREEN_BASE DW 0
COMPLETE_FLAG DW 0
 INCLUDE \LIBRARY\ASMCOP\SETUPTBL.ASC
INTNUMB  DB 0Ch
PORTNUMB DW 03F8h
IRQ  DB 010h   ;(IRQ 4)
SYNCH  DB 0
INSTRT  DW 0
INEND  DW 0
OUTSTRT  DW 0
OUTEND  DW 0
INBUFF  DB 257 DUP('I')
OUTBUFF  DB 257 DUP('O')
OUTCOUNT DW 0
OLDOFF  DW 0
OLDBAS  DW 0
SCLRNBUF DB 20 DUP (' ')
;
;
 PUBLIC SCALE
SCALE PROC FAR
 PUSH BP    ;store BP on the stack
 MOV BP,SP   ;and point BP to top of stack
 PUSH DS    ;store
 PUSH ES    ;base registers
 PUSH CS    ;start
 PUSH CS    ;into
 POP ES    ;8080
 POP DS    ;mode
 MOV CS:COMPLETE_FLAG,0  ;set no reading
 STI    ;turn on interupts
 MOV CX,-1   ;set count
SCLE4:  MOV AX,350   ;set inner loop
SCLE1: DEC AX    ;wait a bit
 JNZ SCLE1   ;then 
 CMP CS:COMPLETE_FLAG,0  ;check flag
 JNE SCLE2   ;for good read
 LOOP SCLE4   ;continue
 MOV BX,[BP].PARM1  ;get pointer to arguement one
 MOV AX,[BP].BASE1  ;get base
 MOV ES,AX   ;set es to base
 MOV BYTE PTR ES:0[BX],'1' ;set timeout error
 JMP SCLE3   ;exit
SCLE2: MOV BX,[BP].PARM1  ;get pointer to arguement one
 MOV AX,[BP].BASE1  ;get base
 MOV ES,AX   ;set es to base
 MOV AL,CS:SCLRNBUF  ;get sign
 MOV ES:1[BX],AL   ;store sign
 MOV AX,WORD PTR CS:SCLRNBUF+9 ;get unit
 MOV ES:8[BX],AX   ;store unit
 MOV AX,WORD PTR CS:SCLRNBUF+12 ;get other
 MOV ES:10[BX],AX  ;store other
 MOV AL,BYTE PTR CS:SCLRNBUF+11 ;get motion
 CMP AL,' '   ;if not space
 JE SCLE8   ;make it
 MOV AL,'M'   ;an M
SCLE8: MOV ES:12[BX],AL  ;store motion
 MOV CX,6   ;set a count
 MOV SI,0   ;and a pointer
SCLE6: MOV AL,SCLRNBUF+2[SI]  ;get data
 CMP AL,' '   ;space?
 JNE SCLE7   ;no
 MOV AL,'0'   ;make zero
SCLE7: MOV ES:2[BX],AL   ;store data
 INC SI    ;set pointers 
 INC BX    ;ahead
 LOOP SCLE6   ;move weight
 MOV BX,[BP].PARM1  ;get pointer to arguement one
 MOV BYTE PTR ES:0[BX],'0' ;set no error
SCLE3: POP ES    ;restore base
 POP DS    ;registers
 POP BP    ;restore BP
 RET    ;exit
SCALE ENDP
;
;
 PUBLIC SCALON
SCALON PROC FAR
 PUSH BP    ;store BP on the stack
 MOV BP,SP   ;and point BP to top of stack
 PUSH DS    ;store
 PUSH ES    ;base registers
 MOV BX,[BP].PARM1  ;get pointer to arguement one
 MOV AX,[BP].BASE1  ;get base for arguement one
 CALL GET_TWO_DIGITS  ;get the number in AX
 MOV BX,80   ;times the line number
 MUL BX    ;then
 MOV DX,AX   ;store
 MOV BX,[BP].PARM2  ;get pointer to arguement two
 MOV AX,[BP].BASE2  ;get base for arguement two
 CALL GET_TWO_DIGITS  ;get the number in AX
 ADD AX,DX   ;equals screen offset
 ADD AX,AX   ;times two for words
 MOV CS:SCREEN_OFFSET,AX  ;store
 CALL GET_SCREEN_BASE  ;get the base from WINDOW
 MOV CS:SCREEN_BASE,AX  ;store
 PUSH CS    ;start
 PUSH CS    ;into
 POP ES    ;8080
 POP DS    ;mode
 READSETUP DONGLE_VALUE  ;read the setup
 SETBAUDRATES DONGLE_VALUE ;set the baud rates
 MOV AL,SYNCHCHAR  ;set default
 MOV SYNCH,AL   ;synch
 CMP COMPORT,0   ;COM 0?
 JE SCO1    ;yes
 MOV INTNUMB,0Bh   ;change setup
 MOV PORTNUMB,02F8h  ;for
 MOV IRQ,08h   ;COM 2
 MOV BX,COMPORT   ;get the
 MOV AL,SYNCHCHAR[BX]  ;synch
 MOV SYNCH,AL   ;and store
SCO1: MOV AL,INTNUMB   ;get the
 MOV AH,35h   ;current
 INT 21h    ;interupt vector
 MOV OLDOFF,BX   ;store
 MOV BX,ES   ;it
 MOV OLDBAS,BX   ;for restoration
 PUSH CS    ;and reset
 POP ES    ;the base register
 LEA DX,COMINT   ;set interupt 0Ch
 MOV AH,25h   ;up for COM0
 MOV AL,INTNUMB   ;by
 INT 21h    ;using the DOS call
ICE: CLI    ;turn off interupts
 MOV DX,PORTNUMB   ;and set DTR,
 ADD DX,4   ;RTS, OUT1, OUT2
 MOV AL,0Fh   ;at the
 OUT DX,AL   ;UART
 MOV DX,021h   ;turn on
 IN AL,DX   ;the IRQ line
 MOV AH,IRQ   ;at
 NOT AH    ;the
 AND AL,AH   ;interupt
 OUT DX,AL   ;controler
 MOV DX,PORTNUMB   ;now read
 INC DX    ;all
 MOV AL,0   ;data
 OUT DX,AL   ;and scrap
 MOV DX,PORTNUMB   ;scrap
 ADD DX,5   ;it
 IN AL,DX   ;so
 MOV DX,PORTNUMB   ;that
 ADD DX,6   ;(MSTAT)
 IN AL,DX   ;no
 MOV DX,PORTNUMB   ;interupts will
 IN AL,DX   ;be
 MOV DX,PORTNUMB   ;pending
 ADD DX,2   ;in
 IN AL,DX   ;the UART
 MOV DX,PORTNUMB   ;set to the
 INC DX    ;UART interupt port
 MOV AL,01h   ;and turn on the input
 OUT DX,AL   ;interupt only
 MOV DX,20h   ;reset the
 MOV AL,20h   ;interupt
 OUT DX,AL   ;controler
 STI    ;then turn on interupts
 MOV CS:COMPLETE_FLAG,0  ;set complete no
 POP ES    ;restore base
 POP DS    ;registers
 POP BP    ;restore BP
 RET    ;exit
SCALON ENDP
;
;
GET_TWO_DIGITS PROC NEAR
 MOV ES,AX   ;set base
 MOV AL,ES:0[BX]   ;get digit
 SUB AL,'0'   ;convert to binary
 MOV AH,AL   ;store copy
 SHL AL,1   ;times 2
 SHL AL,1   ;times 4
 ADD AL,AH   ;times 5
 SHL AL,1   ;times 10
 ADD AL,ES:1[BX]   ;plus next digit
 SUB AL,'0'   ;in binary
 MOV AH,0   ;as sixteen bits
 RET    ;return
GET_TWO_DIGITS ENDP
;
;
 PUBLIC SCALOFF
SCALOFF PROC FAR
 PUSH BP    ;store BP on the stack
 MOV BP,SP   ;and point BP to top of stack
 MOV BX,[BP].PARM1  ;get pointer to arguement one
 PUSH DS    ;store
 PUSH ES    ;base registers
 PUSH CS    ;start
 PUSH CS    ;into
 POP ES    ;8080
 POP DS    ;mode
 CLI    ;turn off interupts
 MOV DX,021h   ;turn off
 IN AL,DX   ;the IRQ line
 XOR AL,IRQ   ;at interupt
 OUT DX,AL   ;controler
 MOV DX,PORTNUMB   ;now read
 INC DX    ;all
 MOV AL,0   ;data
 OUT DX,AL   ;and scrap
 MOV DX,PORTNUMB   ;scrap
 ADD DX,5   ;it
 IN AL,DX   ;so
 MOV DX,PORTNUMB   ;that
 ADD DX,6   ;(MSTAT)
 IN AL,DX   ;no
 MOV DX,PORTNUMB   ;DATA will
 IN AL,DX   ;be
 MOV DX,PORTNUMB   ;pending
 ADD DX,2   ;in
 IN AL,DX   ;the UART
 MOV DX,PORTNUMB   ;set to the
 INC DX    ;UART interupt port
 MOV AL,0h   ;and off all
 OUT DX,AL   ;interupts
 MOV DX,20h   ;reset the
 MOV AL,20h   ;interupt
 OUT DX,AL   ;controler
 MOV DX,OLDOFF   ;set interupt 0Ch
 MOV AX,OLDBAS   ;up
 MOV DS,AX   ;as
 MOV AH,25h   ;it
 MOV AL,CS:INTNUMB  ;was by
 INT 21h    ;using the DOS call
 STI    ;then turn on interupts
 POP ES    ;restore base
 POP DS    ;registers
 POP BP    ;restore BP
 RET    ;exit
SCALOFF ENDP
;
;
COMINT: STI    ;turn interupts back on
 PUSH AX    ;push
 PUSH BX    ;all
 PUSH CX    ;registers
 PUSH DX    ;onto
 PUSH SI    ;the
 PUSH DI    ;stack
 PUSH DS    ;put the base
 PUSH ES    ;registers onto the stack
 MOV AX,CS   ;set to the
 MOV DS,AX   ;local DSEG
 MOV ES,AX   ;area
 MOV AL,20h   ;reset the
 MOV DX,20h   ;Programable
 OUT DX,AL   ;interupt controler
GETINT: MOV DX,PORTNUMB   ;read
 ADD DX,2   ;the
 IN AL,DX   ;interupt ID register
 TEST AL,01h   ;and if no interupt is pending
 JNZ ENDINT   ;then exit
 TEST AL,02h   ;check for
 JZ GI1    ;character
 CALL SEND_CHARACTER  ;to send
 JMP GETINT   ;and repeat
GI1: TEST AL,04h   ;check for
 JZ GI2    ;character
 CALL GET_CHARACTER  ;to be received
 JMP GETINT   ;and repeat
GI2: TEST AL,08h   ;check for
 JZ GI3    ;anomilies (phone ring,
 CALL STATUS_CHANGE  ;hang up, etc.)
GI3: JMP GETINT   ;repeat until all done
ENDINT: POP ES    ;restore
 POP DS    ;all
 POP DI    ;registers
 POP SI    ;from
 POP DX    ;the
 POP CX    ;stack
 POP BX    ;in reverse
 POP AX    ;order
 IRET    ;return from interupt
;
;
SEND_CHARACTER PROC NEAR
 MOV BX,OUTSTRT   ;get pointer to next character
 CMP BX,OUTEND   ;and see if one is there
 JE SC1    ;if not, we are done
 MOV DX,PORTNUMB   ;get the port number
 MOV AL,OUTBUFF[BX]  ;and the character, and write
 OUT DX,AL   ;character to port
 INC BX    ;set pointer ahead
 AND BX,0FFh   ;with 256 word wrap
 MOV OUTSTRT,BX   ;and put it back
 DEC OUTCOUNT   ;decrease characters left
SC1: RET    ;exit
SEND_CHARACTER ENDP
;
;
GET_CHARACTER PROC NEAR
 MOV DX,PORTNUMB   ;read the
 IN AL,DX   ;input character
 MOV BX,INSTRT   ;the get the buffer pointer
 MOV INBUFF[BX],AL  ;and put character into buffer
 INC BX    ;set pointer ahead
 AND BX,0FFh   ;with 256 character wrap
 MOV INSTRT,BX   ;and put it back
 CMP AL,SYNCH   ;if this is a synch character
 JNE GC1    ;then do
 CALL SCALE_IN   ;the scale work
GC1: RET    ;exit
GET_CHARACTER ENDP
;
;
STATUS_CHANGE PROC NEAR
 MOV DX,PORTNUMB   ;read
 ADD DX,5   ;the line
 IN AL,DX   ;status bits
 MOV DX,PORTNUMB   ;to
 ADD DX,6   ;eliminate
 IN AL,DX   ;errors
 RET    ;exit
STATUS_CHANGE ENDP
;
;
SCALE_IN PROC NEAR
 MOV DI,OFFSET SCLRNBUF  ;clear the buffer
 MOV AL,' '   ;to
 MOV CX,15   ;all
 REP STOSB   ;blanks
 MOV CL,WEIGHTLEN  ;get the length of weight
 MOV CH,0   ;as word
 MOV AL,WEIGHTPOS  ;and offset of weight
 MOV AH,0   ;as
 MOV SI,AX   ;word
 MOV DI,8   ;DI = maximum weight length + 1
 SUB DI,CX   ;make pointer to buffer
SI2: MOV AL,INBUFF[SI]  ;get byte of weight
 CMP AL,'-'   ;check for floating sign
 JNE SI3    ;and
 MOV AL,' '   ;set sign
 MOV SCLRNBUF,'-'  ;if found
SI3: CMP AL,' '   ;if space
 JE SI3B    ;continue
 CMP AL,'0'   ;if not
 JL SI3A    ;numeric
 CMP AL,'9'   ;then
 JLE SI3B   ;set no
SI3A: MOV CS:COMPLETE_FLAG,0  ;screen update
SI3B: MOV SCLRNBUF[DI],AL  ;store in screen buffer
 INC SI    ;set to next
 INC DI    ;positions
 LOOP SI2   ;and repeat
 MOV CL,UNITLEN   ;get the unit length
 MOV CH,0   ;as a word
 CMP CX,0   ;if not used
 JE SI4B    ;do next
 MOV AL,UNITPOS   ;get unit of measure
 MOV AH,0   ;as a word
 MOV SI,AX   ;that is source
 MOV DI,9   ;set the answer spot
 MOV BX,0   ;set subscript
SI4A: MOV AL,INBUFF[SI]  ;get data
 CMP AL,BYTE PTR UNITCHECK[BX] ;is it okay?
 JE SI4C    ;yes
 MOV AL,'?'   ;change
SI4C: MOV SCLRNBUF[DI],AL  ;move data
 INC SI    ;set pointers
 INC DI    ;ahead
 INC BX    ;reset BX
 LOOP SI4A   ;and repeat
 CMP SCLRNBUF+9,'K'  ;if kilograms
 JNE SI14X1   ;then
 MOV SCLRNBUF+10,'G'  ;make full word
 JMP SHORT SI4B   ;and continue
SI14X1: CMP SCLRNBUF+9,'L'  ;pounds
 JNE SI4B   ;then
 MOV SCLRNBUF+10,'B'  ;make full word
SI4B: MOV CL,OTHERLEN   ;get the OTHER length
 MOV CH,0   ;as a word
 CMP CX,0   ;if not used
 JE SI5B    ;do next
 MOV AL,OTHERPOS   ;get OTHER
 MOV AH,0   ;as a word
 MOV SI,AX   ;that is source
 MOV DI,12   ;set the answer spot
 MOV BX,0   ;set subscript
SI5A: MOV AL,INBUFF[SI]  ;get data
 CMP AL,BYTE PTR OTHERCHECK[BX] ;is it okay?
 JE SI5C    ;yes
 MOV AL,'?'   ;change
SI5C: MOV SCLRNBUF[DI],AL  ;move data
 INC SI    ;set pointers
 INC DI    ;ahead
 INC BX    ;reset BX
 LOOP SI5A   ;and repeat
SI5B: MOV CL,SIGNLEN   ;get the SIGN length
 CMP CL,0   ;if not used
 JE SI6B    ;do next
 MOV AL,SIGNPOS   ;get SIGN position
 MOV AH,0   ;as a word
 MOV SI,AX   ;that is source
 CMP CL,1   ;single byte?
 JE SI6C    ;yes
 MOV AX,WORD PTR INBUFF[SI] ;get sign
 CMP AX,WORD PTR SIGNCHECK ;check sign
 JMP SHORT SI6D   ;continue
SI6C: MOV AL,BYTE PTR INBUFF[SI] ;get sign
 CMP AL,BYTE PTR SIGNCHECK ;check sign
SI6D: JE SI6B    ;sign okay
 MOV SCLRNBUF,'-'  ;set sign
SI6B: MOV CL,MOTIONLEN  ;get the MOTION length
 CMP CL,0   ;if not used
 JE SI7B    ;do next
 MOV AL,MOTIONPOS  ;get MOTION position
 MOV AH,0   ;as a word
 MOV SI,AX   ;that is source
 CMP CL,1   ;single byte?
 JE SI7C    ;yes
 MOV AX,WORD PTR INBUFF[SI] ;get MOTION
 CMP AX,WORD PTR MOTIONCHECK ;check MOTION
 JMP SHORT SI7D   ;continue
SI7C: MOV AL,BYTE PTR INBUFF[SI] ;get MOTION
 CMP AL,BYTE PTR MOTIONCHECK ;check MOTION
SI7D: JE SI7B    ;MOTION okay
 MOV SCLRNBUF+11,12h  ;set MOTION
SI7B: MOV INSTRT,0   ;reset the buffer pointer
 INC CS:COMPLETE_FLAG  ;set read ready
 CMP CS:COMPLETE_FLAG,2  ;read read?
 JL SIEX    ;do not put on screen
 AND CS:COMPLETE_FLAG,0FFh ;ensure no wrap
 MOV AX,SCREEN_BASE  ;set ES
 MOV ES,AX   ;to the screen base
 MOV DI,SCREEN_OFFSET  ;set DI to the screen
 MOV BX,0   ;offset
 MOV CX,14   ;and CX as a count
SI1: MOV AL,SCLRNBUF[BX]  ;get data byte
 MOV ES:0[DI],AL   ;put on the screen
 ADD DI,2   ;set di to next screen location
 INC BX    ;set bx to next data byte
 LOOP SI1   ;repeat for all characters
SIEX: PUSH CS    ;restore the
 POP ES    ;ES base register
 RET    ;exit
SCALE_IN ENDP
;
;
CSEG ENDS
 END
