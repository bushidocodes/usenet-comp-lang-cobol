        IDENTIFICATION DIVISION.
        PROGRAM-ID. MONEY.

        01  RECORD-10.
            05 DIG1  PIC X(09) VALUE 'ONE '.
            05 DIG2  PIC X(09) VALUE 'TWO '.
            05 DIG3  PIC X(09) VALUE 'THREE '.
            05 DIG4  PIC X(09) VALUE 'FOUR'.
            05 DIG5  PIC X(09) VALUE 'FIVE'.
            05 DIG6  PIC X(09) VALUE 'SIX '.
            05 DIG7  PIC X(09) VALUE 'SEVEN '.
            05 DIG8  PIC X(09) VALUE 'EIGHT '.
            05 DIG9  PIC X(09) VALUE 'NINE'.
            05 DIG10 PIC X(09) VALUE 'TEN '.
            05 DIG11 PIC X(09) VALUE 'ELEVEN '.
            05 DIG12 PIC X(09) VALUE 'TWELVE '.

            05 DIG15 PIC X(09) VALUE 'FIFTEEN '.

            05 DIG17 PIC X(09) VALUE 'SEVENTEEN'.
            05 DIG18 PIC X(09) VALUE 'EIGHTTEEN'.
            05 DIG19 PIC X(09) VALUE 'NINETEEN '.
      * 01  TABEL-RECORD-10  VALUE
      *    'ONE      TWO      THREE     FOUR     FIVE     SIX       SEVE
      *    'N    EIGHT    NINE     TEN      ELEVEN   TWELVE   THIRTEEN F
      *    'OURTEEN FIFTEEN SIXTEEN SEVENTEENEIGHTEEN NINETEEN '.
        01  TABEL-RECORD-10 REDEFINES RECORD-10.
           05 MONEY-10 OCCURS 19 TIMES.
                10 MONEY-WORD-10  PIC X(09).
        01  TABEL-RECORD-20  VALUE
           'TWENTY THIRTY FOURTY FIFTY  SIXTY  SEVENTYEIGHTY NINETY '.
           05 MONEY-20 OCCURS 8  TIMES.
                10 MONEY-WORD-20  PIC X(07).
        01  F-DOLLARS.
            05 F-MONEY      PIC 9(06) VALUE ZEROES.
            05 F-CENT       PIC V99   VALUE ZEROES.
        01  F-SUB           PIC 9(02) VALUE ZEROES.
        01  F-COUNT         PIC 9     VALUE ZEROES.
        01  F-GO-TO         PIC 9     VALUE ZEROES.
        01  MONEY-STRING.
            05 F1           PIC X(09)  VALUE SPACES.
            05 F1A          PIC X(08)  VALUE 'HUNDRED'.
            05 F2           PIC X(07)  VALUE SPACES.
            05 F3           PIC X(09)  VALUE SPACES.
            05 F3A          PIC X(09)  VALUE 'THOUSAND'.
            05 F4           PIC X(07)  VALUE SPACES.
            05 F4A          PIC X(08)  VALUE 'HUNDRED '.
            05 F5           PIC X(07)  VALUE SPACES.
            05 F6           PIC X(09)  VALUE SPACES.
            05 F6A          PIC X(07)  VALUE 'DOLLARS'.
            05 F7           PIC .99    VALUE '.00' .
            05 F7A          PIC X(05)  VALUE 'CENTS'.
        01  MONEY-LINE      PIC X(80) VALUE SPACES.
       PROCEDURE DIVISION.
            MOVE SPACES TO MONEY-STRING.
            ACCEPT  F-DOLLARS.
            INSPECT F-MONEY TALLYING F-COUNT FOR LEADING ZERO.
            DISPLAY 'F-COUNT:  ' F-COUNT
            ADD 1 TO F-COUNT GIVING F-GO-TO.
            DISPLAY 'F-GO-TO:  ' F-GO-TO.
            GO TO
                  PARA-1
                  PARA-2
                  PARA-3
                  PARA-4
                  PARA-5
                  PARA-6
                  DEPENDING ON F-GO-TO.
              PERFORM PARA-FINAL.
       PARA-6.
           MOVE  F-MONEY(6:1) TO F-SUB
           MOVE  MONEY-WORD-10(F-SUB) TO F6.
           MOVE  'DOLLARS'  TO F6A.
           GO TO PARA-FINAL.
       PARA-5.
           MOVE F-MONEY(5:2) TO F-SUB.
           MOVE  'DOLLARS'  TO F6A
           IF F-SUB = 00
              MOVE SPACES TO F5 F6
           ELSE
             IF F-SUB <= 19
               MOVE  MONEY-WORD-10(F-SUB) TO F6
           ELSE
               IF F-SUB(2:1) NOT = 0
                  PERFORM PARA-6A
                  PERFORM PARA-6
               ELSE
                  PERFORM PARA-6A
               END-IF
             END-IF
            END-IF.
           GO TO PARA-FINAL.
       PARA-4.
           MOVE  F-MONEY(4:1) TO F-SUB
           IF
              F-SUB = 0
              MOVE SPACES TO F4 F4A
           ELSE
              MOVE  MONEY-WORD-10(F-SUB) TO F4
              MOVE  'HUNDRED'  TO F4A.
           PERFORM PARA-5.
       PARA-3.
           MOVE F-MONEY(3:1) TO F-SUB.
           MOVE  MONEY-WORD-10(F-SUB) TO F3.
           MOVE  'THOUSAND' TO F3A.
           PERFORM PARA-4.
       PARA-3A.
            MOVE F-MONEY(2:1) TO F-SUB
            MOVE MONEY-WORD-20(F-SUB - 1) TO F2.
       PARA-2.
           MOVE F-MONEY(2:2) TO F-SUB.
               MOVE  'THOUSAND' TO F3A
           IF F-SUB = 00
               MOVE  SPACES TO F3  F2
               PERFORM PARA-4
           ELSE
           IF F-SUB <= 19
               MOVE  MONEY-WORD-10(F-SUB) TO F3
               PERFORM PARA-4
           ELSE
               IF F-SUB(2:1) NOT = 0
                  PERFORM PARA-3A
                  PERFORM PARA-3
               ELSE
                  PERFORM PARA-3A
                  PERFORM PARA-4
               END-IF
            END-IF.
       PARA-1.
            MOVE F-MONEY(1:1) TO F-SUB
            MOVE MONEY-WORD-10(F-SUB) TO F1.
            MOVE  'HUNDRED'  TO F1A.
            PERFORM PARA-2.
       PARA-6A.
            MOVE F-MONEY(5:1) TO F-SUB
            MOVE MONEY-WORD-20(F-SUB - 1) TO F5.
       PARA-FINAL.
            IF F-CENT  = .00
               MOVE '00' TO F7(2:2)
               MOVE 'CENT' TO F7A
            ELSE
             IF F-CENT = .01
               MOVE 'CENT' TO F7A
               MOVE '01' TO F7(2:2)
             ELSE
               MOVE F-CENT      TO F7
               MOVE 'CENTS' TO F7A
            END-IF
            END-IF.
            MOVE SPACE TO F7(1:1).
            STRING
                 F1  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F1A DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F2  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F3  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F3A DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F4  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F4A DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F5  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F6  DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 F6A DELIMITED BY SPACE
                 ' ' DELIMITED BY SIZE
                 'AND' DELIMITED BY SIZE
                 ' ' DELIMITED BY SIZE
                 F7  DELIMITED  BY SIZE
                 '/100 ' DELIMITED BY SIZE
                 F7A DELIMITED BY SPACE
                     INTO MONEY-LINE
           END-STRING.
           DISPLAY MONEY-LINE.
           STOP RUN.
