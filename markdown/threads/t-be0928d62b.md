      * THIS PROGRAM WILL REPLACE NUMERIC CHARACTERS WITH ALPHABETIC

      **************************************************************
       ENVIRONMENT DIVISION.
       DATA DIVISION.
       WORKING-STORAGE SECTION.
       01  HARD-UNITS          PIC X(45) VALUE
            'ONE  TWO  THREEFOUR FIVE SIX  SEVENEIGHTNINE '.
       01  TABLE-UNITS REDEFINES HARD-UNITS.
          05  UNITS    OCCURS 9 TIMES.
             10  UNITS-VALUE       PIC X(5).
       01  HARD-TENNERS  PIC X(63)  VALUE
                      'TEN    TWENTY THIRTY FORTY  FIFTY  SIXTY  SEVENTY
      -               'EIGHTY NINETY '.
       01 TABLE-TENNERS REDEFINES HARD-TENNERS.
          05  TENNERS OCCURS 9 TIMES.
              10  TENS-VALUE     PIC X(7).
       01  HARD-COMPOUNDS      PIC X(81)  VALUE
                 'ELEVEN   TWELVE   THIRTEEN FOURTEEN FIFTEEN  SIXTEEN
      -          'SEVENTEENEIGHTEEN NINETEEN '.
       01  TABLE-COMPOUND REDEFINES HARD-COMPOUNDS.
          05  COMPOUNDS   OCCURS 9 TIMES.
              10  COMP-VALUE      PIC X(9).
       01 INPUT-TABLE VALUE ZEROES.
          05  ENTRIES OCCURS 4 TIMES.
              10  H       PIC 9.
              10  T       PIC 9.
              10  U       PIC 9.
       01  CHAR-CNTR.
           05 WORD-CNTR            PIC S9(2) COMP VALUE +0.
           05 LAST-CHAR            PIC S9(2) COMP VALUE +1.
           05 SPACE-REMAINING       PIC S9(2) COMP VALUE +0.
       01  SUB1                    PIC S9(2) COMP VALUE +0.
       01  SUB                     PIC S9(2) COMP VALUE +0.
       01  SUB-OUT                 PIC S9(2) COMP VALUE +1.
       01  MORE-INQUIRES           PIC X     VALUE 'Y'.
       01  SLOT                    PIC X(15)   VALUE SPACES.
       01  DISPLAY-TABLE.
           05  DISPLAY-LINE OCCURS 3 TIMES PIC X(80) VALUE SPACES.
       PROCEDURE DIVISION.
       000-MAIN-LINE.
           DISPLAY                'ENTER TWELVE NUMERIC CHARACTERS, CAN NOT BE
      -           ' NOT BE ALL ZEROES: TO EXIT - PRESS (N)'.
      *    IF INPUT-TABLE IS NOT NUMERIC
      *     DISPLAY 'NOT NUMERIC VALUE! CONTINUE OR PRESS (N) TO EXIT'.
      *     PERFORM 000-MAIN-LINE.
           PERFORM 1000-PROCESS UNTIL MORE-INQUIRES = 'N'.
           STOP RUN.
       1000-PROCESS.
           ACCEPT INPUT-TABLE
           IF INPUT-TABLE = 'N'
             MOVE 'N' TO MORE-INQUIRES.
           IF INPUT-TABLE IS NOT NUMERIC
      *     DISPLAY 'NOT NUMERIC VALUE!'.
            PERFORM 000-MAIN-LINE.
           PERFORM VARYING SUB1 FROM 1 BY 1 UNTIL SUB1 > 4
                 PERFORM EVALUATE-ENTRIES
           END-PERFORM.
           PERFORM CHECK-TABLE
           DISPLAY 'GET YOUR RESULT:' DISPLAY DISPLAY-TABLE
           DISPLAY 'ARE THERE MORE INQUIRES (Y/N)?'.
           ACCEPT MORE-INQUIRES.
           DISPLAY 'ENTER TWELVE NUMERIC CHARACTERS:'
           INITIALIZE DISPLAY-TABLE SLOT WORD-CNTR.
           MOVE +1 TO LAST-CHAR.
       EVALUATE-ENTRIES.
           EVALUATE TRUE ALSO TRUE ALSO TRUE
               WHEN H(SUB1) = 0 ALSO T(SUB1) = 0 ALSO U(SUB1) > 0
                   PERFORM TRIPLET-1
               WHEN H(SUB1) = 0 ALSO T(SUB1) > 0 ALSO U(SUB1) = 0
                   PERFORM TRIPLET-2
               WHEN H(SUB1) > 0 ALSO T(SUB1) = 0 ALSO U(SUB1) = 0
                   PERFORM TRIPLET-3
               WHEN H(SUB1) = 0 ALSO T(SUB1) > 0 ALSO U(SUB1) > 0
                   PERFORM TRIPLET-4
               WHEN H(SUB1) > 0 ALSO T(SUB1) > 0 ALSO U(SUB1) = 0
                   PERFORM TRIPLET-5
               WHEN H(SUB1) > 0 ALSO T(SUB1) = 0 ALSO U(SUB) > 0
                   PERFORM TRIPLET-6
               WHEN H(SUB1) > 0 ALSO T(SUB1) > 0 ALSO U(SUB1) > 0
                   PERFORM TRIPLET-7
               WHEN OTHER CONTINUE
           END-EVALUATE.
       TRIPLET-1.
           PERFORM REGULAR-UNITS
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       TRIPLET-2.
           PERFORM REGULAR-TENS
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       TRIPLET-3.
           PERFORM REGULAR-HUNDS
           MOVE 'HUNDRED' TO SLOT
           PERFORM COUNT-LINE
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       TRIPLET-4.
           IF T (SUB1) = 1
               PERFORM TENS-ONE
           ELSE
               PERFORM REGULAR-TENS
               PERFORM REGULAR-UNITS
           END-IF
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       TRIPLET-5.
           PERFORM REGULAR-HUNDS
           MOVE 'HUNDRED' TO SLOT
           PERFORM COUNT-LINE
           PERFORM REGULAR-TENS
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       TRIPLET-6.
           PERFORM REGULAR-HUNDS
           MOVE 'HUNDRED' TO SLOT
           PERFORM COUNT-LINE
           PERFORM REGULAR-UNITS.
       TRIPLET-7.
           PERFORM REGULAR-HUNDS
           MOVE 'HUNDRED' TO SLOT
           PERFORM COUNT-LINE
           IF T (SUB1) = 1
               PERFORM TENS-ONE
           ELSE
               PERFORM REGULAR-TENS
               PERFORM REGULAR-UNITS
           END-IF
           PERFORM MOVE-CNT-TXT
           PERFORM COUNT-LINE.
       REGULAR-TENS.
           MOVE T (SUB1) TO SUB
           MOVE TENS-VALUE (SUB) TO SLOT
           PERFORM COUNT-LINE.
       REGULAR-UNITS.
           MOVE U (SUB1) TO SUB
           MOVE UNITS-VALUE (SUB) TO SLOT
           PERFORM COUNT-LINE.
       REGULAR-HUNDS.
           MOVE H (SUB1) TO SUB
           MOVE UNITS-VALUE (SUB) TO SLOT.
           PERFORM COUNT-LINE.
       TENS-ONE.
           MOVE U (SUB1) TO SUB
           MOVE COMP-VALUE (SUB) TO SLOT.
           PERFORM COUNT-LINE.
       MOVE-CNT-TXT.
           IF SUB1 = 1 MOVE 'BILLION' TO SLOT.
           IF SUB1 = 2 MOVE 'MILLION' TO SLOT.
           IF SUB1 = 3 MOVE 'THOUSAND' TO SLOT.
           IF SUB1 = 4 PERFORM LAST-IF.
       CHECK-TABLE.
          IF INPUT-TABLE = 000000000000
              PERFORM 000-MAIN-LINE.
       LAST-IF.
          IF INPUT-TABLE = 000000000001
              MOVE 'DOLLAR' TO SLOT
          ELSE
              MOVE 'DOLLARS' TO SLOT
          END-IF.
       COUNT-LINE.
           INSPECT SLOT TALLYING WORD-CNTR FOR CHARACTERS
             BEFORE INITIAL SPACES
           SUBTRACT LAST-CHAR FROM 80 GIVING SPACE-REMAINING
           IF SPACE-REMAINING < WORD-CNTR
              ADD 1 TO SUB-OUT
              MOVE SLOT TO DISPLAY-LINE(SUB-OUT)(1: )
              MOVE ZEROES TO LAST-CHAR
              ADD 2 WORD-CNTR TO LAST-CHAR
           ELSE
              IF SPACE-REMAINING = WORD-CNTR
                 MOVE SLOT TO DISPLAY-LINE(SUB-OUT)(LAST-CHAR: )
              ELSE
                 MOVE SLOT TO DISPLAY-LINE(SUB-OUT)(LAST-CHAR: )
                 ADD 1 WORD-CNTR TO LAST-CHAR
              END-IF
           END-IF.
           MOVE ZEROES TO WORD-CNTR.
