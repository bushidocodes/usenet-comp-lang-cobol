We have developed some utilities in-house which makes things much
easier for COBOL programmers to work in VB.

Included in this is a program which will populate a VB screen from a
UDT which links to a DLL which gives full acces to MF COBOL files.

It also writes all of the event handling code to allow the Enter key
to move forward through the code and the ESC key to move back, thus
allowing for much faster data entry than the standard VB methods.

Another of the programs will create the UDT records from a COBOL copy
file, and also the load-UDT and save-UDT sections required for moving
the COBOL file data into the VB DLL.

I may be prepared to release these if there is any interest.

The following shows the input to and output from the code generator:
This is all of the code which is required to do this properly as you
would in COBOL. This example is, of course, trivial. Just thnk,
though. how much time is saved of there are 50 fields on the screen!

The program works equally well if you wish to use unbound processing
with a Jet or ODBC Database.

Don'r say you did not know....................


INPUT ********************************************

Private Sub Tx1Account_No_Change()

End Sub
Private Sub Tx1Name_Change()

End Sub

Private Sub Tx1Addr_1_Change()

End Sub
Private Sub Tx1Addr_2_Change()

End Sub

END OF INPUT **********************************

OUTPUT*******************************************

Dim Esc, F1, F2, F3, F4, F5, F6, F7, F8, F9, F10, F11, F12 As Integer
Dim Cr_Up, Cr_Down, Cr_Left, Cr_Right, Home_Key, Print_Scr,Close_Key
As Integer
Private Sub Clear_Screen()
Tx1Account_No.Text = ""
Tx1Account_No.Refresh
Tx1Name.Text = ""
Tx1Name.Refresh
Tx1Addr_1.Text = ""
Tx1Addr_1.Refresh
Tx1Addr_2.Text = ""
Tx1Addr_2.Refresh
End Sub
Private Sub Load_Screen()
If Len(SLMAS.Account_No) > 0 Then
Tx1Account_No = SLMAS.Account_No
Tx1Account_No.Refresh
End If
If Len(SLMAS.Name) > 0 Then
Tx1Name = SLMAS.Name
Tx1Name.Refresh
End If
If Len(SLMAS.Addr_1) > 0 Then
Tx1Addr_1 = SLMAS.Addr_1
Tx1Addr_1.Refresh
End If
If Len(SLMAS.Addr_2) > 0 Then
Tx1Addr_2 = SLMAS.Addr_2
Tx1Addr_2.Refresh
End If
End Sub
Private Sub Save_Screen()
If Len(Tx1Account_No.Text) > 0 Then
SLMAS.Account_No = Tx1Account_No.Text
End If
If Len(Tx1Name.Text) > 0 Then
SLMAS.Name = Tx1Name.Text
End If
If Len(Tx1Addr_1.Text) > 0 Then
SLMAS.Addr_1 = Tx1Addr_1.Text
End If
If Len(Tx1Addr_2.Text) > 0 Then
SLMAS.Addr_2 = Tx1Addr_2.Text
End If
End Sub
Private Sub Tx1Account_No_Keyup(KeyCode As Integer, Shift As Integer)
If KeyCode = Esc Then
Tx1Account_No.SetFocus
Else
If KeyCode <> 13 Then
Tx1Account_No.SetFocus
Else
SLMAS.Account_No = Tx1Account_No.Text
Tx1Name.SetFocus
End If
End If
End Sub
Private Sub Tx1Name_Keyup(KeyCode As Integer, Shift As Integer)
If KeyCode = Esc Then
Tx1Account_No.SetFocus
Else
If KeyCode <> 13 Then
Tx1Name.SetFocus
Else
SLMAS.Name = Tx1Name.Text
Tx1Addr_1.SetFocus
End If
End If
End Sub
Private Sub Tx1Addr_1_Keyup(KeyCode As Integer, Shift As Integer)
If KeyCode = Esc Then
Tx1Name.SetFocus
Else
If KeyCode <> 13 Then
Tx1Addr_1.SetFocus
Else
SLMAS.Addr_1 = Tx1Addr_1.Text
Tx1Addr_2.SetFocus
End If
End If
End Sub
Private Sub Tx1Addr_2_Keyup(KeyCode As Integer, Shift As Integer)
If KeyCode = Esc Then
Tx1Addr_1.SetFocus
Else
If KeyCode <> 13 Then
Tx1Addr_2.SetFocus
Else
SLMAS.Addr_2 = Tx1Addr_2.Text
Tx1Addr_2.SetFocus
End If
End If
End Sub
Private Sub Tx1Account_No_Keypress(keyAscii As Integer)
Suppress_Beep
End Sub
Private Sub Tx1Name_Keypress(keyAscii As Integer)
Suppress_Beep
End Sub
Private Sub Tx1Addr_1_Keypress(keyAscii As Integer)
Suppress_Beep
End Sub
Private Sub Tx1Addr_2_Keypress(keyAscii As Integer)
Suppress_Beep
End Sub
Private Sub Suppress_Beep ()
If keyAscii = 13 Or keyAscii = 27 Then
keyAscii = 0
End If
End Sub
Private Sub Load_Keys()

Home_Key = 36
End_Key = 35

Enter_Key = 13
End Sub
