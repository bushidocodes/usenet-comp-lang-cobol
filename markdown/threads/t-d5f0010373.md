PIC S9(4) SIGN LEADING SEPARATE VALUE +1000 X'4EF1F0F0F0' "+1000"
PIC S9(4) SIGN LEADING SEPARATE VALUE -1000 X'60F1F0F0F0' "-1000"

PIC 9(4) VALUE 1000 X'F1F0F0F0' "1000"
PIC S9(4) SIGN TRAILING SEPARATE VALUE +1000 X'F1F0F0F04E' "1000+"
PIC S9(4) SIGN TRAILING SEPARATE VALUE -1000 X'F1F0F0F060' "1000-"

PIC 9(4) VALUE 1000 X'F1F0F0F0' "1000"
PIC S9(4) SIGN LEADING VALUE +1000 X'C1F0F0F0' "A000"
PIC S9(4) SIGN LEADING VALUE -1000 X'D1F0F0F0' "J000"

PIC 9(4) VALUE 1000 X'F1F0F0F0' "1000"
PIC S9(4) SIGN TRAILING VALUE +1000 X'F1F0F0C0' "100{"
PIC S9(4) SIGN TRAILING VALUE -1000 X'F1F0F0D0' "100}"

The default is SIGN TRAILING.
