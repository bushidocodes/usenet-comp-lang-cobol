Please check with Bob England of England Technical Services, but I'm
almost positive that COBOL RPC from ETS can help you accomplish
exactly what you need to do.  COBOL RPC provides a server daemon which
can launch an instance of a server-based COBOL application.  It then
provides the ability to pass a parameter block across TCP/IP.  The
parameter block can be passed from a client machine to a server, vice
versa or from one server to another.

ETS also has versions which work identically on UNIX servers or
Windows NT servers.  I carbon copied Bob on this e-mail so that he is

This is very basic, but there you go:

#include <stdio.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netdb.h>
#include <netinet/in.h>

#ifdef DEBUG
#define D(x) fprintf(stderr, x);
#else
#define D(x)
#endif

void doit(int);
void err_sys(char *);
int writen(int, char *, int);
int readn(int, char *, int);
int readline(int, char *, int);
char *myservice = "cpcsrvd";

main(argc, argv)
int argc;
char *argv[];
{
    int sockfd, newsockfd;
    int child_pid;
    struct servent *myservent;
    struct sockaddr_in srv_addr;
    short srv_port;
    struct sockaddr_in cli_addr;
    short cli_port;
    int cli_len;

    D("myservent")
    myservent = getservbyname(myservice, "tcp");
    if (myservent == NULL)
 err_sys("getservbyname error");
    if ((srv_port = ntohs(myservent->s_port)) == 0)
 err_sys("port error");
    D("socket")
    if ((sockfd = socket(AF_INET, SOCK_STREAM, 0)) < 0)
 err_sys("socket error");
    bzero((char *)&srv_addr, sizeof(srv_addr));
    srv_addr.sin_family = AF_INET;
    srv_addr.sin_addr.s_addr = htonl(INADDR_ANY);
    srv_addr.sin_port = htons(srv_port);
    D("bind")
    if (bind(sockfd, (struct sockaddr *)&srv_addr, sizeof(srv_addr))  < 0)

 err_sys("bind error");
    D("listen")
    if (listen(sockfd, 5) < 0)
 err_sys("listen error");

    for (;;) {
 cli_len = sizeof(cli_addr);
 D("accept")
 newsockfd = accept(sockfd, (struct sockaddr *)&cli_addr, &cli_len);
      /* blocks */
 if (newsockfd < 0)
     err_sys("accept error");
 D("fork")
 if ((child_pid = fork()) < 0)
     err_sys("fork error");
 else if (child_pid == 0) {
     close(sockfd);   /* child */
     doit(newsockfd);   /* process the request */
     /* NOT REACHED */
 }
 D("parent")
 close(newsockfd);   /* parent */
    }
}

void err_sys(s)
char *s;
{
    fprintf(stderr, "cpcsrvd: %s\n", s);
    perror("cpcsrvd");
    exit(1);
}

#define MAXARGV 20
void doit(sockfd)
int sockfd;
{
    char *av[MAXARGV+1];
    char cobol[BUFSIZ+1];
    int n;

    D("doit")
    if ((n = recv(sockfd, cobol, BUFSIZ, 0)) < 1)
 err_sys("recv error");
/* first message we receive is name of cobol object file to execute */
/* this works for RM/COBOL, you'd have to adapt it to other COBOLs */
/* after invocation, the COBOL program has an open connection on file
descriptor 0 */
    cobol[n] = '\0';
    av[0] = "runcobol";
    av[1] = cobol;
    av[2] = "-k";
    av[3] = NULL;
    if (dup2(sockfd, 0) < 0)
 err_sys("dup2 error");
    close(sockfd);
    if (execv("/usr/bin/runcobol", av) < 1)
 err_sys("execv error");
    exit(1);
}

int writen(fd, ptr, nbytes)
register int fd;
register char *ptr;
register int nbytes;
{
    int nleft;
    int nwritten;

    nleft = nbytes;
    while (nleft > 0) {
 nwritten = write(fd, ptr, nleft);
 if (nwritten < 0)   /* error */
     return(nwritten);
 nleft -= nwritten;
 ptr += nwritten;
    }
    return(nbytes - nleft);
}

int readn(fd, ptr, nbytes)
register int fd;
register char *ptr;
register int nbytes;
{
    int nleft;
    int nread;

    nleft = nbytes;
    while (nleft > 0) {
 nread = read(fd, ptr, nleft);
 if (nread < 0)
     return(nread);   /* error */
 else if (nread == 0)
     break;    /* EOF */
 nleft -= nread;
 ptr += nread;
    }
    return(nbytes - nleft);
}

int readline(fd, ptr, maxlen)
register int fd;
register char *ptr;
register int maxlen;
{
    int n;
    int rc;
    char c;

    for (n = 1; n < maxlen; n++) {
 if ((rc = read(fd, &c, 1)) == 1) {
     *ptr++ = c;
     if (c == '\n')
  break;
 } else if (rc == 0) {
     if (n == 1)
  return(0); /* EOF, no data read */
     else
  break;  /* EOF, some data was read */
 } else
     return(-1);  /* error */
    }

    *ptr = 0;
    return(n);
}
