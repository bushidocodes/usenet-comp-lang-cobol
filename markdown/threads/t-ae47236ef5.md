Basically if you can read the fields in hex you can translate them  (if they
are really comp-3).

Packed decimal fields only contain numeric hex values apart from the
rightmost half byte which is a sign.  Normally  F or C denote positive
values and D denotes negative values.  Eg X'01234F' and X'01234C' are 1234
whereas X'01234D' is 1234-.

Regards
Rick

You can try this code. It is written in FoxPro, but I guess that it is
understandable. You pass two parameters to the function, the comp-3 string
and the number of decimals. It returns a string with decimal point.

The function is not tested, so take care....

func comp3
 parameter m_str, antdec
 private m_str, m_data, m_cnt, m_luku, m_ub, m_lb, m_uc, m_lc
 m_ub  = 0
 m_lb  = 0
 m_uc  = ""
 m_lc  = ""
 m_cnt = len(m_str)
 m_data = ""
 for i = 1 to m_cnt-1
  asctkn = asc(subs(m_str,i,1))
  m_ub = int(asctkn/16)
  m_lb = asctkn-(m_ub * 16)
  m_uc = chr(48+m_ub)
  m_lc = chr(48+m_lb)
  m_data = m_data + m_uc + m_lc
 next
 asctkn = asc(subs(m_str,i,1))
 m_ub = int(asctkn/16)
 m_lb = asctkn-(m_ub * 16)
 m_uc = chr(48+m_ub)
 m_lc = chr(48+m_lb)
 m_data = m_data + m_uc

 if asctkn % 2 = 0    && even
  m_data = '+' + m_data
 else
  m_data = '-' + m_data
 endif
return left(m_data, len(m_data)-antdec) + '.' + right(m_data, antdec)

Because COMP-3 is non-standard, its implementation is defined
by the implementor. The most common form is that defined by IBM.
This is the format you learned in "the dim and distant past."

As for translating the above, I have not worked out all the details
since there is an insufficient range of examples. For example,
no negative numbers or decimal fractions. For the examples
given, here is the code. The rest is up to you! :-)

/* fltpnt.c */

#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <math.h>

int main ( void );
void sftostr ( unsigned char * sf, char * s );

int main ( void )
{
 unsigned char a_float[] = { 0x00, 0xb0, 0x01, 0x8e };
 char s[10];

 sftostr ( a_float, s );
 printf ( "%s\n", s );
 return 0;
}

/* strange float to string */
void sftostr ( unsigned char * sf, char * s )
{
 double d = 0.0;
 int exponent;
 d = (double) sf[2] * 65536;
 d += (double) sf[1] * 256;
 d += (double) sf[0];
 d /= 4096.0;
 if (sf[3] & 0x80)
  d += 2048.0;
 exponent = ( sf[3] & 0x7F ) - 12;
 d *= pow ( 2.0, (double) exponent );
 d /= 100.0;
 sprintf ( s, "%f", d );
}

/* fltpnt.c */

#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <math.h>

int main ( void );
void sftostr ( unsigned char * sf, char * s, double scale );

int main ( void )
{
unsigned char a_float[6][4] = {{ 0x00, 0xb0, 0x01, 0x8e },
        { 0x00, 0xa0, 0x3e, 0x8d },
        { 0x00, 0x60, 0x06, 0x8e },
        { 0x00, 0x10, 0x3d, 0x8e },
        { 0x00, 0xd0, 0x04, 0x8e },
        { 0x00, 0x40, 0x1c, 0x8f } };
size_t x;
char s[10];
for ( x = 0; x < 6; x++ )
{
  sftostr ( a_float[x], s, 100.0 );
  printf ( "%s\n", s );
}
return 0;
}

/* strange float to string */
void sftostr ( unsigned char * sf, char * s, double scale )
{
double d = 0.0;
int exponent;
/* extract fraction */
d = (double) sf[2] * 65536.0;
d += (double) sf[1] * 256.0;
d += (double) sf[0];
d /= 16777216.0;
/* adjust */
d += 0.5;
/* adjust for sign of number - magnitude */
if (!(sf[3] & 0x80))
  d *= -1.0;
/* get exponent */
exponent = ( sf[3] & 0x3F );
/* adjust for sign of exponent */
if (sf[3] & 0x40)
  exponent *= -1;
/* adjust the number */
d *= pow ( 2.0, (double) exponent );
/* scale the number */
d /= scale;
/* convert to string */
sprintf ( s, "%f", d );
}

COMP-3 is ALWAYS SIGNED, therefor the physical size is computed as

6/2 = 3 bytes.
