names: procedure options (main);
   declare name character (50) varying;
   declare k fixed binary;

   do forever;
      get edit (name) (L);
      name = trim(name);
      do until (k = 0); /* Ensure only 1 blank between names. */
         k = index(name, '  ');
         if k > 0 then name = substr(name, 1, k) || trim(substr(name, k+2));
      end;
      k = searchr(name, ' ');
      if k = 0 then put skip list (name);
      else put skip list (substr(name, k+1) || ' ' || substr(name, 1, k-1));
   end;

end names;
