Here's a simplified version of a perl hack which we have used for this
purpose. Assumes that:

* target format is GNT in the current directory
* copy statements have double quotes surrounding the file name (")

----begin-inclusion-----------------------------------------------------
#!/usr/local/bin/perl

require 5.000;

die "Usage: cobdepend files\n" unless @ARGV;
die "cobdepend: \$COBCPY not set or null\n"
unless defined $ENV{'COBCPY'} && length $ENV{'COBCPY'};

$_ = $ENV{'COBCPY'};
s/^:/.:/;
s/:$/:./;
s/::/:.:/g;
my @path = split /:/;
my ($last_target, $target, $copy, $full) = ('');

while (<>)
{
chomp;
next unless ($copy) = /^(?:\t|.{6}\s)\s*copy\s+"(.*?)"/i;
unless ($copy =~ m!/!)
{
for (@path)
{
$full = "$_/$copy";
$copy = $full, last if -f $full;
}
}

warn "cobdepend: can't find $copy\n" unless -f $copy;
($target = ($last_target = $ARGV)) =~ s/\.(cbl|cob)/.gnt/i
unless $ARGV eq $last_target;

print "$target: $copy\n";
}
----end-----------------------------------------------------------------
