```
Here are the variables that I set.  I'm running NT4.0 SP3...

@echo off
echo Configuring Micro Focus Cobol System Environment...

rem **********************************************************
rem * Configure Micro Focus Cobol System Environment:        *
rem **********************************************************
set path=c:\mfcb32w\exedll;%path%
set cobdir=c:\mfcb32w\lbr;c:\mfcb32w\exedll;%cobdir%
set cobcpy=c:\mfcb32w\source;%cobcpy%
set mforg=c:\c-cubed\source\cta\mforgwin
set lib=c:\mfds32w\lib\gui;c:\mfcb32w\lib;%lib%

rem **********************************************************
rem * Check for Windows NT startup:                          *
rem **********************************************************
if "%computername%"=="THLUCAS_P400" start /B %1 %2 %3 %4 %5

rem **********************************************************
rem * Check for Windows 95 startup:                          *
rem **********************************************************
if "%computername%"=="THLUCAS_LAP" %1 %2 %3 %4 %5

exit
```
