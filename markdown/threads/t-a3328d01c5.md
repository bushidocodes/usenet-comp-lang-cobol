If 16bit windows...
     $ set ans85 noosvs mf defaultbyte"00"
       IDENTIFICATION DIVISION.
       PROGRAM-ID. play.
       SPECIAL-NAMES.
                call-convention 3 is WINAPI.
       DATA DIVISION.
       WORKING-STORAGE SECTION.
       01 mfptr procedure-pointer.

       procedure division.
           set mfptr to entry "mmsystem"
           call WINAPI "sndPlaySound" using
                   by reference "c:\windows\chimes.wav" & x"0",
                   by value 0 size 2.
           display return-code

If 32bit windows...
      $set defaultbyte"00" case
       special-names.
            call-convention 74 is WINAPI.

       working-storage section.

       78  SND-SYNC          value h"0000".
       78  SND-ASYNC         value h"0001".
       78  SND-NODEFAULT     value h"0002".
       78  SND-MEMORY        value h"0004".
       78  SND-LOOP          value h"0008".
       78  SND-NOSTOP        value h"0010".

       01 mfdata.
           03 ptr-to-mmsystem  procedure-pointer.
           03 badptr procedure-pointer.

       procedure division.

           set badptr to entry "guess_this_is_a_pointer"
           set ptr-to-mmsystem to entry "winmm"

           if badptr equals ptr-to-mmsystem
               display "Multimedia sound DLL not present"
               stop run
           end-if

           call winapi "sndPlaySoundA" using
               by reference "c:\win95\media\chimes.wav" & x"0",
               by value SND-ASYNC size 2
           end-call

           if return-code not equal 0
               display "Failed"
           end-if
