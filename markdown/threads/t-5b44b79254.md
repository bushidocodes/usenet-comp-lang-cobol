Procedure Division.
*>--lecture imprimante par defaut (read default printer)

by value lpsz2
by value lpsz3
by value lpsz4
by value lnDefPrinter

*>--lecture imprimantes disponibles (read all the printers)
set lpsz1 to address of devSection
set lpsz3 to address of devDefault
set lpsz4 to address of lstDevices

by value lpsz1
by value 0 size 4
by value lpsz3
by value lpsz4
by value lnLstDevices
returning lonLstDevices end-call
*>--positionne imprimante par defaut (set default printer)
if returnBuffer<> newBuffer
set lpszNewBuffer to address of newBuffer
call WINAPI "WriteProfileString" using
by value lpszSection
by value lpszEntry
by value lpszNewBuffer
returning apiRetour end-call
*>--flush win.ini
call WINAPI "WriteProfileString" using
by value 0 size 4
by value 0 size 4
by value 0 size 4
returning apiRetour end-call
end-if

move PFONT-USE-FONT-DIALOG to Open-With-Options

by value Open-With-Options
returning Printer-RetCode end-call
*>--
if Printer-RetCode= 0
*>--demande hdc et infos sur imprimante (gets hdc and printer
information)
move Printer-Struct-Size to Size-Of-Info-Structure
call "PC_WIN_PRINTER_INFO" using Printer-Handle
Printer-Info-Struct
returning Printer-RetCode end-call
*>--demande resolution de l'imprimante (gets printer definition)

call WINAPI "GetDeviceCaps" using
by value Windows-Printer-Hdc
by value logPixelsY size 2
returning vRes end-call
*>--demande nom de la fonte (gets font name)
initialize fontTypeFace
set lpszFace to address of fontTypeFace
move lnTypeFace to lnFace
call WINAPI "GetTextFace" using
by value Windows-Printer-Hdc
by value lnFace
by value lpszFace
returning lnFace end-call
*>--demande infos sur la fonte (gets font information)
set lptm to address of textMetrics
call WINAPI "GetTextMetrics" using
by value Windows-Printer-Hdc
by value lptm
returning bool end-call
*>--calcule taille de la fonte (compute font size)
compute taille rounded=
((tmHeight - tmIntLeading) * 72) / vRes
*>--
initialize cParam
move non to cItalic cUnder cStruck cBold
move fontTypeFace to cfontTypeFace
move taille to cTaille
if tmItalic= 255
move oui to cItalic
end-if
if tmUnder= 255
move oui to cUnder
end-if
if tmStruck= 255
move oui to cStruck
end-if
if tmWeight> 600
move oui to cBold
end-if
end-if
exit program.
stop run.
