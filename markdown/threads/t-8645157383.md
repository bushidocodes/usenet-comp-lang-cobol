#!/bin/env python

import os, sys, string, re

from optparse import OptionParser

LinuxOptions = "--fcompiler=cob"
WindowsOptions = "--compiler=mingw32 --fcompiler=cob"
supportedExtensions = [".cbl", ".cob"]

makeTemplate = """PROGRAM        = %(programName)s

COBCOMPILER    = cob
FFLAGS         = %(compilerOptions)s
LDFLAGS        =
LIBS           =

all: $(PROGRAM)

%(makeRules)s

ALL_SOURCE = %(allSource)s

clean:
    rm -rf *.int *.idy *.lst *.gnt

mrproper:
    rm -rf $(PROGRAM)

"""

def findProgramName(sourceFilename):

    line = sourceFilename.replace('.',' ')
    comps = line.split()
    if len(comps)==2:
        return comps[0].strip()
   
    return None

def findDepends(sourceFilename):

    sourceFile = file(sourceFilename, "r")
    lines = sourceFile.readlines()
    sourceFile.close()

    moduleDepends = []
    p=re.compile('copy')
    for line in lines:
        if p.search(line.lower())!=None:
            q=re.compile('"[a-zA-Z0-9./-_]+"')
            r = q.search( line.lower())
            if r:
                if r.group().strip('"') not in ('cbl_copy_file'):
                    moduleDepends.append(r.group().strip('"'))
    return moduleDepends

def isSupportedExtension(entry):
    for extension in supportedExtensions:
        if entry.find(extension)!=-1:
            return True
    return False

def findSupportedExtension(entry):
    for extension in supportedExtensions:
        if entry.find(extension)!=-1:
            return extension
    return ""


def generateMakefile(targetDir, inputProgramName, compilerOptions,
programSuffix):

    dirContents = os.listdir(targetDir)
    currDir = os.getcwd()
    os.chdir(targetDir)

    sourceFiles = []
    moduleSourceFiles = []
    objectFiles = {}
    programName = ""

    if sys.platform == "win32":
        CompilerOptions = WindowsOptions
    else:
        CompilerOptions = LinuxOptions

    for dirEntry in dirContents:
        if os.path.isfile(dirEntry):
            if isSupportedExtension(dirEntry):
                extension = findSupportedExtension(dirEntry)
                sourceFiles.append(dirEntry)
                testProgramName = findProgramName(dirEntry)
                depends = findDepends(dirEntry)
                objectFiles[dirEntry.split(".")[0]] = [depends, extension]

                if testProgramName!=None:
                    depends = findDepends(dirEntry)
                    programName = testProgramName

    makeRules = []

    if inputProgramName!="[Auto]":
        programName = inputProgramName

    for object in objectFiles.keys():

        extension = objectFiles[object][1]
        makeRules.append(object+".int: "+object+extension+" "+"
".join([obj for obj in objectFiles[object][0]]))

    for object in objectFiles.keys():

        extension = objectFiles[object][1]
        makeRules.append(object+".gnt: "+object+extension+" "+"
".join([obj for obj in objectFiles[object][0]]))
       
    makeRules.append("")
   
    makeRules.append("%.int : %.cbl")
    makeRules.append("\t$(COBCOMPILER) $(FFLAGS) $<")
    makeRules.append("%.gnt : %.cbl")
    makeRules.append("\t$(COBCOMPILER) $(FFLAGS) -u $<")

    makeRules.append("")   
   
    programName="prog"
    makeRules.append(programName+": "+"int gnt")

    makeRules.append("int:"+" ".join([object+".int" for object in
objectFiles.keys()]))
    makeRules.append("gnt:"+" ".join([object+".gnt" for object in
objectFiles.keys()]))

    sourceFilesString = " ".join(sourceFiles)
    moduleSourceFilesString = " ".join(moduleSourceFiles)
    makeRuleString = "\n".join(makeRules)

    makeFile = file("makefile", "w")
    makeFile.write(makeTemplate % {
        "programName":programName,
        "makeRules":makeRuleString,
        "allSource":sourceFilesString,
        "moduleSource":moduleSourceFilesString,
        "compilerOptions":compilerOptions,
        "programSuffix":programSuffix,
        "CompilerOptions":CompilerOptions}
    )
    makeFile.close()

    os.chdir(currDir)

def printUsage():
    pass

if __name__ == "__main__":

    # Display user interface if tkinter is installed and no arguments
are given, otherwise
    # run as command line tool.

    parser = OptionParser()
    parser.add_option("-t", "--target-dir", dest="targetDir",
default=".", help="Generate makefiles in DIR.", metavar="DIR")
    parser.add_option("-a", "--app-name", dest="appName",
default="[Auto]", help="Application name.")
    parser.add_option("-c", "--compiler-options",
dest="compilerOptions", default="-P -C \"CONSTANT LIEU 'BXL' \"",
help="Compiler options.")

    if sys.platform == "win32":
        parser.add_option("-s", "--program-suffix", default="exe",
dest="programSuffix", help="Executable suffix.")
    else:
        parser.add_option("-s", "--program-suffix", default="",
dest="programSuffix", help="Executable suffix.")

    parser.add_option("-g", "--gui", action="store_true", dest="useGui",
default=False, help="Show user interface. Requires Tkinter and Tcl/Tk.")

    (options, args) = parser.parse_args()

    if options.useGui:
        root = Tk()
        appWindow = AppWindow(root, targetDir=options.targetDir,
appName=options.appName, compilerOptions=options.compilerOptions,
programSuffix=options.programSuffix)
        root.mainloop()
        sys.exit(0)
    else:
        print "Generating makefiles.",
        generateMakefile(options.targetDir, options.appName,
options.compilerOptions, options.programSuffix)
        print "Done."
