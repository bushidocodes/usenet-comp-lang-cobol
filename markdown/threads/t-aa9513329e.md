unsigned addcrc(unsigned Crc, unsigned char *b, int i)
{
while (0 < i--)
Crc = (crctab[((Crc >> 8) & 255)] ^ (Crc << 8) ^ *b++);

return (Crc & 0x0000ffff);
}


unsigned calccrc(unsigned Crc, unsigned char *b, int i)
{
register unsigned char *cp;
register int k;

while (0 < i)
{
for (k = 0, cp = b; k < i; k++, cp++)
if (*cp <= CTLZ)
break;

if (k > 0)
{
Crc = addcrc(Crc, b, k);

if (++k >= i)
break;
}
else
k++;

while (*++cp <= CTLZ && k < i)
k++;

i -= k;
b += k;
}

return Crc;
}


unsigned updcrc(unsigned cp, unsigned Crc)
{
return (0x0000ffff & (crctab[((Crc >> 8) & 255)] ^ (Crc << 8) ^ p));
}


unsigned crcfinish(unsigned Crc)
{
/*
* Call updcrc twice, passing it a character of hex 00 each time, to
* flush out the last 16 bits from the CRC calculation, and return the
* result as the value of this function.
*/
return (updcrc(0, updcrc(0, Crc)));
}
