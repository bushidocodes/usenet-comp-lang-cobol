01  CALLED-FIELD.
    03  CF-BYTE             OCCURS 64
                           INDEXED CFX
                               PIC X(01).

01  CALLED-LENGTH              PIC 9(09)  COMP SYNC.
/
PROCEDURE DIVISION           USING CALLED-FIELD
                                   CALLED-LENGTH.
MAINLINE
    .
** guard against overflow
*
    IF CALLED-LENGTH > 64
       EXIT PROGRAM
    END-IF
*
** get our actual field length
*
    SET CFX TO CALLED-LENGTH
    PERFORM UNTIL CF-BYTE (CFX) > SPACE
               OR CFX < 1
        SET CFX DOWN BY 1
    END-PERFORM

    IF CFX < 1
       EXIT PROGRAM
    END-IF

    SET FIELD-LENGTH TO CFX
*
** look for a lowercase character - if found, exit
*
    SET CFX TO 1
    PERFORM UNTIL CFX > FIELD-LENGTH
       IF CF-BYTE (CFX) >= 'a' and <= 'z'
          EXIT PROGRAM
       END-IF
       SET CFX UP BY 1
    END-PERFORM
*
** initialize
*
    MOVE SPACES                TO TEST-FIELD
                                  OUTPUT-FIELD
    MOVE 1                     TO OUTPUT-POS
*
** leading spaces?
*
    SET CFX TO 1

    PERFORM UNTIL CF-BYTE (CFX) > SPACE
               OR CFX > FIELD-LENGTH
       SET CFX UP BY 1
    END-PERFORM
*
** convert the field, piece by piece
*
    PERFORM UNTIL CFX > FIELD-LENGTH
       PERFORM CONVERT-FRAGMENT
    END-PERFORM
*
** we done, baby
*
    MOVE OUTPUT-FIELD            TO CALLED-FIELD

    EXIT PROGRAM
    .
/
CONVERT-FRAGMENT
    .
    SET TEST-POS TO CFX
*
** find the length of the next chunk
*
    PERFORM UNTIL CF-BYTE (CFX) = SPACE OR ',' OR '-'
               OR CFX > FIELD-LENGTH
       SET CFX UP BY 1
    END-PERFORM

    SET TEST-LEN TO CFX
    SUBTRACT TEST-POS FROM TEST-LEN
*
** test-field is where we do the actual work
*
    MOVE CALLED-FIELD (TEST-POS:TEST-LEN) TO TEST-FIELD

    PERFORM CONVERT-TEST-FIELD
*
** output-field is our finished product
*
    MOVE TEST-FIELD (1:TEST-LEN) TO OUTPUT-FIELD (OUTPUT-POS:TEST-LEN)

    ADD TEST-LEN TO OUTPUT-POS
*
** our next field is either a space, comma, period or dash
*
    MOVE CF-BYTE (CFX)  TO OUTPUT-FIELD (OUTPUT-POS:1)
    IF OUTPUT-FIELD (OUTPUT-POS:1) = ','
       ADD 2 TO OUTPUT-POS
    ELSE
       ADD 1 TO OUTPUT-POS
    END-IF
*
** this will solve the multiple space problem
*
    SET CFX UP BY 1
    PERFORM UNTIL CF-BYTE (CFX) > SPACE
               OR CFX > FIELD-LENGTH
       SET CFX UP BY 1
    END-PERFORM
    .
/
CONVERT-TEST-FIELD
    .
** lots of special cases
*
    IF TEST-FIELD (1:1) = '(' OR '"' OR ''''
       MOVE TEST-FIELD (1:1)   TO OUTPUT-FIELD (OUTPUT-POS:1)
       ADD 1 TO OUTPUT-POS
       MOVE TEST-FIELD (2:TEST-LEN) TO TEST-FIELD
       SUBTRACT 2 FROM TEST-LEN
       MOVE 'Y'                TO PAREN-SW
    END-IF

    PERFORM CHECK-ALL-UPPER

    IF OUTPUT-POS > 1
       PERFORM CHECK-ALL-LOWER
    END-IF

    PERFORM CHECK-MC-MAC

    IF TEST-FIELD (1:1) NUMERIC
       PERFORM NUMBER-RTN
    END-IF
*
** special case for addresses like "King's Rd"; otherwise it's "O'Reilly" and
** the like.  For everything else we add 32 to the ASCII value
*
    PERFORM UNTIL TEST-FIELD (TEST-POS:1) = SPACE
               OR TEST-POS > 64
       IF TEST-FIELD (TEST-POS:1) >= 'A' AND <= 'Z'
          MOVE TEST-FIELD (TEST-POS:1) TO ASCII-CHAR
          ADD VALUE-32 TO ASCII-VAL
          MOVE ASCII-CHAR      TO TEST-FIELD (TEST-POS:1)
          ADD 1 TO TEST-POS
       ELSE
       IF TEST-FIELD (TEST-POS:3) = '''S'
          ADD 1 TO TEST-POS
       ELSE
          ADD 2 TO TEST-POS
       END-IF END-IF
    END-PERFORM

    IF PAREN-SW = 'Y'
       MOVE 'N'                TO PAREN-SW
       ADD 1 TO TEST-LEN
    END-IF
    .
/
CHECK-ALL-UPPER
    .
    SET AUX TO 1
    SEARCH ALL-UPPER-ENTRIES
       END
          CONTINUE
       WHEN ALL-UPPER-ENTRIES (AUX) = TEST-FIELD
          MOVE 65              TO TEST-POS
    END-SEARCH
    .
/
CHECK-ALL-LOWER
    .
    SET ALX TO 1
    SEARCH ALL-LOWER-ENTRIES
       END
          CONTINUE
       WHEN ALL-LOWER-ENTRIES (ALX) = TEST-FIELD
          MOVE 1               TO TEST-POS
    END-SEARCH
    .
/
CHECK-MC-MAC
    .
    IF TEST-FIELD (1:2) = 'MC'
       MOVE 'Mc'               TO TEST-FIELD (1:2)
       ADD 2 TO TEST-POS
    ELSE
    IF TEST-FIELD (1:3) = 'MAC'
       SEARCH ALL MAC-ENTRIES
          END
             CONTINUE
          WHEN MAC-ENTRY (MACX) = TEST-FIELD (1:TEST-LEN)
             MOVE 'Mac'        TO TEST-FIELD (1:3)
             ADD 3 TO TEST-POS
       END-SEARCH
    END-IF END-IF
    .
/
NUMBER-RTN
    .
    PERFORM UNTIL TEST-FIELD (TEST-POS:1) NOT NUMERIC
               OR TEST-POS > 64
       ADD 1 TO TEST-POS
    END-PERFORM
*
** once we are past the numbers then we must convert 21ST to 21st and the like
*
    IF TEST-FIELD (TEST-POS:3) = 'ST'
       MOVE 'st'      TO TEST-FIELD (TEST-POS:2)
       ADD 2 TO TEST-POS
    ELSE
    IF TEST-FIELD (TEST-POS:3) = 'ND'
       MOVE 'nd'      TO TEST-FIELD (TEST-POS:2)
       ADD 2 TO TEST-POS
    ELSE
    IF TEST-FIELD (TEST-POS:3) = 'RD'
       MOVE 'rd'      TO TEST-FIELD (TEST-POS:2)
       ADD 2 TO TEST-POS
    ELSE
    IF TEST-FIELD (TEST-POS:3) = 'TH'
       MOVE 'th'      TO TEST-FIELD (TEST-POS:2)
       ADD 2 TO TEST-POS
    ELSE
       ADD 1 TO TEST-POS
    END-IF END-IF END-IF END-IF
    .
