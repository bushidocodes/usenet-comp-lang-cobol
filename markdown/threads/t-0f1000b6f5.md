and i  get  all  rows   one  by one and  shows all

by the way, this routine was made to get one by one

always getting the Next  record



           EXEC  SQL

            SELECT top 1

             CAST(REPLACE(YEAR(tit_dtemissao), ' ', 0) +

                  REPLACE(STR(MONTH(tit_dtemissao), 2), ' ', 0) +

                  REPLACE(STR(DAY(tit_dtemissao), 2), ' ', 0)   +

                  REPLACE(STR(tit_numero, 6), ' ', 0) +

                  REPLACE(STR(tit_desd, 2), ' ', 0)  AS
                  NUMERIC(16,0) ) , *

             INTO

                   :sql-chavedata , :regtit

             FROM   Teste.dbo.titulo WITH (NOLOCK)

             WHERE

             CAST(REPLACE(YEAR(tit_dtemissao), ' ', 0) +

                  REPLACE(STR(MONTH(tit_dtemissao), 2), ' ', 0) +

                  REPLACE(STR(DAY(tit_dtemissao), 2), ' ', 0)   +

                  REPLACE(STR(tit_numero, 6), ' ', 0) +

                  REPLACE(STR(tit_desd, 2), ' ', 0)  AS NUMERIC(16,0) )

                       > :sql-chavedata  and

                          tit_empresa  = :sql-empresa

            ORDER  BY  tit_dtemissao, tit_numero, tit_desd

           END-EXEC
