   SELECT D002-COM       ASSIGN TO "COM1"
                   ORGANIZATION IS RECORD SEQUENTIAL
                   ACCESS IS SEQUENTIAL
                   FILE STATUS IS D002-STATUS.
       DATA DIVISION.
       FILE SECTION.
       FD  D002-COM.
       01  D002-COM-RECORD             PIC X.
       WORKING-STORAGE SECTION.

       PROCEDURE DIVISION OS2API.
           CALL X'91' USING WS00-RESULT, WS00-FUNCTION, WS00-CALL-CMD.
           STOP RUN.
       1000-PROCESS-COMM-PORT.
      *    DISPLAY '1000-PROCESS-COMM-PORT' AT 0201.
           CALL "CBL_GET_KBD_STATUS" USING KEY-STATUS
                         RETURNING RET-CODE.

           READ D002-COM
                AT END
                   DISPLAY 'AT END'.
