      $set ans85
      *****************************************************************
       identification division.
       program-id. ajudaven.

       environment division.
       configuration section.
       special-names. decimal-point is comma
                      crt status is ws-escape.
       input-output section.
       file-control.
             copy   arqven.ext.
       data division.
       file section.
       copy  vendedor.ext.

       working-storage section.
       01  ws-avanca               pic x         value spaces.
       copy   dthrstat.ext.

       screen section.
           copy  ajuda.ss.
           copy  mensa.ss.

       procedure division.
       inicio section.
           open     input arqven
           if        ws-status not = "00" and "05"
                     move     "ARQVEN  " to ws-nomarq
                     perform  erro-status1
                     close     arqven
                     display  spaces at 0101
                     exit program.

           move  1  to ws-indt
           perform salva-tela
           display tela-ajuda
           move    spaces  to  vnd-nome
           start   arqven key  is not less than vnd-nome
           move    zeros      to ws-coord
           move    99         to ws-cod03
           move    spaces  to  ws-avanca
           perform until ws-cod03  = zeros
             if      ws-avanca = spaces
                     move "A"  to ws-avanca
             else
                     move    spaces  to  ws-avanca
                     accept  ws-avanca  at 2479    with auto
             end-if
             evaluate  true
              when  ws-avanca = ("A" or "a") and
                                (ws-status not = "10" and "46")
                                or ws-avanca = "p" or "P"
                   if  ws-avanca = "P" or "p"
                       move  2  to ws-indt
                       perform salva-tela
                       display tela-mensa
                       move    spaces  to  vnd-nome
                       move    99      to  ws-status
                       perform until ws-status  = "00" or "02"
                          move  spaces  to  vnd-nome
                          accept  vnd-nome  at 2220
                          start   arqven key  is not less than vnd-nome
                          if      ws-status not = "00" and "02"
                                  display
                   "O Nome que foi Digitado nao esta Cadastrado <Enter>"
                                  at 2020
                                  display  vnd-nome   at 2220
                          end-if
                       end-perform
                       move  2  to ws-indt
                       perform recupera-tela
                   end-if
                 move    06      to  ws-lin
                 perform  until ws-lin > 20
                    perform leitura-vendedor-next
                    if   ws-status = "10" or "46"
                         if      ws-lin < 20
                                 move    ws-lin to ws-lin ws-linh
                                 move    20     to ws-col
                                 perform varying ws-lin from ws-linh
                                                 by 1 until ws-lin > 20
                                    display ws-brancos at ws-coord with
                                                          size 59
                                 end-perform
                         end-if
                         move  99      to ws-lin
                         move  zeros   to ws-status
                         move  spaces  to  vnd-nome
                    else
                         move    20      to  ws-col
                         display vnd-chave   at ws-coord
                         move    27      to  ws-col
                         display "-"         at ws-coord
                         move    29      to  ws-col
                         display vnd-nome    at ws-coord
                         add     1  to  ws-lin
                    end-if
                 end-perform
              when  ws-avanca = ("R" or "r")
                 move    20      to  ws-lin
                 perform  until ws-lin < 06
                    perform leitura-vendedor-prev
                    if   ws-status = "10" or "46"
                         if      ws-lin > 06
                                 move    ws-lin to ws-lin ws-linh
                                 move    20     to ws-col
                                 perform varying ws-lin from ws-linh
                                                 by -1 until ws-lin < 06
                                    display ws-brancos at ws-coord with
                                                          size 59
                                 end-perform
                         end-if
                         move 01       to ws-lin
                         move  zeros   to ws-status
                         move  "zzzz"  to  vnd-nome
                    else
                         move    20      to  ws-col
                         display vnd-chave   at ws-coord
                         move    27      to  ws-col
                         display "-"         at ws-coord
                         move    29      to  ws-col
                         display vnd-nome    at ws-coord
                         subtract  1  from   ws-lin
                    end-if
                 end-perform
             end-evaluate
           end-perform
           move  1  to ws-indt
           perform recupera-tela.
       fim-processamento.
           close      arqven
           exit program.

       leitura-vendedor-next.
           read      arqven next with ignore lock at end
                     move  "10" to  ws-status.
           if        ws-status not = "00" and "02" and "10" and
                                     "23" and "46"
                     move    "ARQVEN  " to ws-nomarq
                     move     "N" to ws-dispmsg
                     perform erro-status1
                     go   fim-processamento.

       leitura-vendedor-prev.
           read      arqven previous ignore lock.
           if        ws-status not = "00" and "02" and "10" and
                                     "23" and "46"
                     move    "ARQVEN  " to ws-nomarq
                     move     "N"       to ws-dispmsg
                     perform erro-status1
                     go   fim-processamento.

           copy      taberros.ext.
           copy      salvale.ext.
